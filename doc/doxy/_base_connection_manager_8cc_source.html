35 /**
36  * On a torus the end and the begin of the axes are connected so you
37  * get a circle. On a circle the distance between two points can't be greater
38  * than half of the circumference.
39  * If the normal distance between two points on one axis is bigger than
40  * half of the size there must be a "shorter way" over the border on this axis
41  */
42 static double dist(double coord1, double coord2, double size)
43 {
44  double difference = fabs(coord1 - coord2);
45  if (difference == 0)
46  // NOTE: event if size is zero
47  return 0;
48  else {
49  ASSERT(size != 0);
50  double dist = FWMath::modulo(difference, size);
51  return std::min(dist, size - dist);
52  }
53 }
54 
55 double sqrTorusDist(Coord c, Coord b, Coord size)
56 {
57  double xDist = dist(c.x, b.x, size.x);
58  double yDist = dist(c.y, b.y, size.y);
59  double zDist = dist(c.z, b.z, size.z);
60  return xDist * xDist + yDist * yDist + zDist * zDist;
61 }

64 void BaseConnectionManager::initialize(int stage)
65 {
66  // BaseModule::initialize(stage);
67 
68  if (stage == 0) {
69  drawMIR = hasPar("drawMaxIntfDist") ? par("drawMaxIntfDist").boolValue() : false;
70 
71  EV_TRACE << "initializing BaseConnectionManager\n";
72 
73  BaseWorldUtility* world = FindModule<BaseWorldUtility*>::findGlobalModule();
74 
75  ASSERT(world != nullptr);
76 
77  playgroundSize = world->getPgs();
78  useTorus = world->useTorus();
79 
80  if (hasPar("sendDirect"))
81  sendDirect = par("sendDirect").boolValue();
82  else
83  sendDirect = false;
84 
85  maxInterferenceDistance = calcInterfDist();
86  maxDistSquared = maxInterferenceDistance * maxInterferenceDistance;
87 
88  // ----initialize node grid-----
89  // step 1 - calculate dimension of grid
90  // one cell should have at least the size of maxInterferenceDistance
91  // but also should divide the playground in equal parts
92  Coord dim((*playgroundSize) / maxInterferenceDistance);
93  gridDim = GridCoord(dim);
94 
95  // A grid smaller or equal to 3x3 would mean that every cell has every
96  // other cell as direct neighbor (if our playground is a torus, even if
97  // not the most of the cells are direct neighbors of each other. So we
98  // reduce the grid size to 1x1.
99  if ((gridDim.x <= 3) && (gridDim.y <= 3) && (gridDim.z <= 3)) {
100  gridDim.x = 1;
101  gridDim.y = 1;
102  gridDim.z = 1;
103  }
104  else {
105  gridDim.x = std::max(1, gridDim.x);
106  gridDim.y = std::max(1, gridDim.y);
107  gridDim.z = std::max(1, gridDim.z);
108  }
109 
110  // step 2 - initialize the matrix which represents our grid
111  NicEntries entries;
112  RowVector row;
113  NicMatrix matrix;
114 
115  for (int i = 0; i < gridDim.z; ++i) {
116  row.push_back(entries); // copy empty NicEntries to RowVector
117  }
118  for (int i = 0; i < gridDim.y; ++i) { // fill the ColVector with copies of
119  matrix.push_back(row); // the RowVector.
120  }
121  for (int i = 0; i < gridDim.x; ++i) { // fill the grid with copies of
122  nicGrid.push_back(matrix); // the matrix.
123  }
124  EV_TRACE << " using " << gridDim.x << "x" << gridDim.y << "x" << gridDim.z << " grid" << endl;
125 
126  // step 3 - calculate the factor which maps the coordinate of a node
127  // to the grid cell
128  // if we use a 1x1 grid every coordinate is mapped to (0,0, 0)
129  findDistance = Coord(std::max(playgroundSize->x, maxInterferenceDistance), std::max(playgroundSize->y, maxInterferenceDistance), std::max(playgroundSize->z, maxInterferenceDistance));
130  // otherwise we divide the playground into cells of size of the maximum
131  // interference distance
132  if (gridDim.x != 1) findDistance.x = playgroundSize->x / gridDim.x;
133  if (gridDim.y != 1) findDistance.y = playgroundSize->y / gridDim.y;
134  if (gridDim.z != 1) findDistance.z = playgroundSize->z / gridDim.z;
135 
136  // since the upper playground borders (at pg-size) are part of the
137  // playground we have to assure that they are mapped to a valid
138  // (the last) grid cell we do this by increasing the find distance
139  // by a small value.
140  // This also assures that findDistance is never zero.
141  const auto epsilon = 0.001;
142  findDistance += Coord(epsilon, epsilon, epsilon);
143 
144  // findDistance (equals cell size) has to be greater or equal
145  // maxInt-distance
146  ASSERT(findDistance.x >= maxInterferenceDistance);
147  ASSERT(findDistance.y >= maxInterferenceDistance);
148  ASSERT(findDistance.z >= maxInterferenceDistance);
149 
150  // playGroundSize has to be part of the playGround
151  ASSERT(GridCoord(*playgroundSize, findDistance).x == gridDim.x - 1);
152  ASSERT(GridCoord(*playgroundSize, findDistance).y == gridDim.y - 1);
153  ASSERT(GridCoord(*playgroundSize, findDistance).z == gridDim.z - 1);
154  EV_TRACE << "findDistance is " << findDistance.info() << endl;
155  }
156  else if (stage == 1) {
157  }
158 }

191 void BaseConnectionManager::checkGrid(BaseConnectionManager::GridCoord& oldCell, BaseConnectionManager::GridCoord& newCell, int id)
192 
193 {
194 
195  // structure to find union of grid squares
196  CoordSet gridUnion(74);
197 
198  // find nic at old position
199  NicEntries& oldCellEntries = getCellEntries(oldCell);
200  NicEntries::iterator it = oldCellEntries.find(id);
201  NicEntries::mapped_type nic = it->second;
202 
203  // move nic to a new position in matrix
204  if (oldCell != newCell) {
205  oldCellEntries.erase(it);
206  getCellEntries(newCell)[id] = nic;
207  }
208 
209  if ((gridDim.x == 1) && (gridDim.y == 1) && (gridDim.z == 1)) {
210  gridUnion.add(oldCell);
211  }
212  else {
213  // add grid around oldPos
214  fillUnionWithNeighbors(gridUnion, oldCell);
215 
216  if (oldCell != newCell) {
217  // add grid around newPos
218  fillUnionWithNeighbors(gridUnion, newCell);
219  }
220  }
221 
222  GridCoord* c = gridUnion.next();
223  while (c != nullptr) {
224  EV_TRACE << "Update cons in [" << c->info() << "]" << endl;
225  updateNicConnections(getCellEntries(*c), nic);
226  c = gridUnion.next();
227  }
228 }

230 int BaseConnectionManager::wrapIfTorus(int value, int max)
231 {
232  if (value < 0) {
233  if (useTorus) {
234  return max + value;
235  }
236  else {
237  return -1;
238  }
239  }
240  else if (value >= max) {
241  if (useTorus) {
242  return value - max;
243  }
244  else {
245  return -1;
246  }
247  }
248  else {
249  return value;
250  }
251 }

253 void BaseConnectionManager::fillUnionWithNeighbors(CoordSet& gridUnion, GridCoord cell)
254 {
255  for (int iz = (int) cell.z - 1; iz <= (int) cell.z + 1; iz++) {
256  int cz = wrapIfTorus(iz, gridDim.z);
257  if (cz == -1) {
258  continue;
259  }
260  for (int ix = (int) cell.x - 1; ix <= (int) cell.x + 1; ix++) {
261  int cx = wrapIfTorus(ix, gridDim.x);
262  if (cx == -1) {
263  continue;
264  }
265  for (int iy = (int) cell.y - 1; iy <= (int) cell.y + 1; iy++) {
266  int cy = wrapIfTorus(iy, gridDim.y);
267  if (cy != -1) {
268  gridUnion.add(GridCoord(cx, cy, cz));
269  }
270  }
271  }
272  }

275 bool BaseConnectionManager::isInRange(BaseConnectionManager::NicEntries::mapped_type pFromNic, BaseConnectionManager::NicEntries::mapped_type pToNic)
276 {
277  double dDistance = 0.0;
278 
279  if (useTorus) {
280  dDistance = sqrTorusDist(pFromNic->pos, pToNic->pos, *playgroundSize);
281  }
282  else {
283  dDistance = pFromNic->pos.sqrdist(pToNic->pos);
284  }
285  return (dDistance <= maxDistSquared);
286 }

288 void BaseConnectionManager::updateNicConnections(NicEntries& nmap, BaseConnectionManager::NicEntries::mapped_type nic)
289 {
290  int id = nic->nicId;
291 
292  for (NicEntries::iterator i = nmap.begin(); i != nmap.end(); ++i) {
293  NicEntries::mapped_type nic_i = i->second;
294 
295  // no recursive connections
296  if (nic_i->nicId == id) continue;
297 
298  bool inRange = isInRange(nic, nic_i);
299  bool connected = nic->isConnected(nic_i);
300 
301  if (inRange && !connected) {
302  // nodes within communication range: connect
303  // nodes within communication range && not yet connected
304  EV_TRACE << "nic #" << id << " and #" << nic_i->nicId << " are in range" << endl;
305  nic->connectTo(nic_i);
306  nic_i->connectTo(nic);
307  }
308  else if (!inRange && connected) {
309  // out of range: disconnect
310  // out of range, and still connected
311  EV_TRACE << "nic #" << id << " and #" << nic_i->nicId << " are NOT in range" << endl;
312  nic->disconnectFrom(nic_i);
313  nic_i->disconnectFrom(nic);
314  }
315  }

324 
325  // create new NicEntry

355 bool BaseConnectionManager::unregisterNic(cModule* nicModule)
356 {
357  ASSERT(nicModule != nullptr);
358 
359  // find nicEntry
360  int nicID = nicModule->getId();
361  EV_TRACE << " unregistering nic #" << nicID << endl;
362 
363  // we assume that the module was previously registered with this CM
364  // TODO: maybe change this to an omnet-error instead of an assertion
365  ASSERT(nics.find(nicID) != nics.end());
366  NicEntries::mapped_type nicEntry = nics[nicID];
367 
368  // get all affected grid squares
369  CoordSet gridUnion(74);
370  GridCoord cell = getCellForCoordinate(nicEntry->pos);
371  if ((gridDim.x == 1) && (gridDim.y == 1) && (gridDim.z == 1)) {
372  gridUnion.add(cell);
373  }
374  else {
375  fillUnionWithNeighbors(gridUnion, cell);
376  }
377 
378  // disconnect from all NICs in these grid squares
379  GridCoord* c = gridUnion.next();
380  while (c != nullptr) {
381  EV_TRACE << "Update cons in [" << c->info() << "]" << endl;
382  NicEntries& nmap = getCellEntries(*c);
383  for (NicEntries::iterator i = nmap.begin(); i != nmap.end(); ++i) {
384  NicEntries::mapped_type other = i->second;
385  if (other == nicEntry) continue;
386  if (!other->isConnected(nicEntry)) continue;
387  other->disconnectFrom(nicEntry);
388  nicEntry->disconnectFrom(other);
389  }
390  c = gridUnion.next();
391  }
392 
393  // erase from grid
394  NicEntries& cellEntries = getCellEntries(cell);
395  cellEntries.erase(nicID);
396 
397  // erase from list of known nics
398  nics.erase(nicID);
399 
400  delete nicEntry;
401 
402  return true;
403 }
