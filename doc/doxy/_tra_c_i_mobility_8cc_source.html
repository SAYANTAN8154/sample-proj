135 void TraCIMobility::handleSelfMsg(cMessage* msg)
136 {
137  if (msg == startAccidentMsg) {
138  getVehicleCommandInterface()->setSpeed(0);
139  simtime_t accidentDuration = par("accidentDuration");
140  scheduleAt(simTime() + accidentDuration, stopAccidentMsg);
141  accidentCount--;
142  }
143  else if (msg == stopAccidentMsg) {
144  getVehicleCommandInterface()->setSpeed(-1);
145  if (accidentCount > 0) {
146  simtime_t accidentInterval = par("accidentInterval");
147  scheduleAt(simTime() + accidentInterval, startAccidentMsg);
148  }
149  }

190 void TraCIMobility::changePosition()
191 {
192  // ensure we're not called twice in one time step
193  ASSERT(lastUpdate != simTime());
194 
195  Coord nextPos = calculateHostPosition(roadPosition);
196  nextPos.z = move.getStartPosition().z;
197 
198  // keep statistics (for current step)
199  currentPosXVec.record(nextPos.x);
200  currentPosYVec.record(nextPos.y);
201 
202  // keep statistics (relative to last step)
203  if (statistics.startTime != simTime()) {
204  simtime_t updateInterval = simTime() - this->lastUpdate;
205 
206  double distance = move.getStartPos().distance(nextPos);
207  statistics.totalDistance += distance;
208  statistics.totalTime += updateInterval;
209  if (speed != -1) {
210  statistics.minSpeed = std::min(statistics.minSpeed, speed);
211  statistics.maxSpeed = std::max(statistics.maxSpeed, speed);
212  currentSpeedVec.record(speed);
213  if (last_speed != -1) {
214  double acceleration = (speed - last_speed) / updateInterval;
215  double co2emission = calculateCO2emission(speed, acceleration);
216  currentAccelerationVec.record(acceleration);
217  currentCO2EmissionVec.record(co2emission);
218  statistics.totalCO2Emission += co2emission * updateInterval.dbl();
219  }
220  last_speed = speed;
221  }
222  else {
223  last_speed = -1;
224  speed = -1;
225  }
226  }
227  this->lastUpdate = simTime();
228 
229  // Update display string to show node is getting updates
230  auto hostMod = getParentModule();
231  if (std::string(hostMod->getDisplayString().getTagArg("veins", 0)) == ". ") {
232  hostMod->getDisplayString().setTagArg("veins", 0, " .");
233  }
234  else {
235  hostMod->getDisplayString().setTagArg("veins", 0, ". ");
236  }
237 
238  move.setStart(Coord(nextPos.x, nextPos.y, move.getStartPosition().z)); // keep z position
239  move.setDirectionByVector(heading.toCoord());
240  move.setOrientationByVector(heading.toCoord());
241  if (this->setHostSpeed) {
242  move.setSpeed(speed);
243  }
244  fixIfHostGetsOutside();
245  updatePosition();
246 }

260 void TraCIMobility::fixIfHostGetsOutside()
261 {
262  Coord pos = move.getStartPos();
263  Coord dummy = Coord::ZERO;
264  double dum;
265 
266  bool outsideX = (pos.x < 0) || (pos.x >= playgroundSizeX());
267  bool outsideY = (pos.y < 0) || (pos.y >= playgroundSizeY());
268  bool outsideZ = (!world->use2D()) && ((pos.z < 0) || (pos.z >= playgroundSizeZ()));
269  if (outsideX || outsideY || outsideZ) {
270  throw cRuntimeError("Tried moving host to (%f, %f) which is outside the playground", pos.x, pos.y);
271  }
272 
273  handleIfOutside(RAISEERROR, pos, dummy, dummy, dum);
274 }

276 double TraCIMobility::calculateCO2emission(double v, double a) const
277 {
278  // Calculate CO2 emission parameters according to:
279  // Cappiello, A. and Chabini, I. and Nam, E.K. and Lue, A. and Abou Zeid, M., "A statistical model of vehicle emissions and fuel consumption," IEEE 5th International Conference on Intelligent Transportation Systems (IEEE ITSC), pp. 801-809, 2002
280 
281  double A = 1000 * 0.1326; // W/m/s
282  double B = 1000 * 2.7384e-03; // W/(m/s)^2
283  double C = 1000 * 1.0843e-03; // W/(m/s)^3
284  double M = 1325.0; // kg
285 
286  // power in W
287  double P_tract = A * v + B * v * v + C * v * v * v + M * a * v; // for sloped roads: +M*g*sin_theta*v
288 
289  /*
290  // "Category 7 vehicle" (e.g. a '92 Suzuki Swift)
291  double alpha = 1.01;
292  double beta = 0.0162;
293  double delta = 1.90e-06;
294  double zeta = 0.252;
295  double alpha1 = 0.985;
296  */
297 
298  // "Category 9 vehicle" (e.g. a '94 Dodge Spirit)
299  double alpha = 1.11;
300  double beta = 0.0134;
301  double delta = 1.98e-06;
302  double zeta = 0.241;
303  double alpha1 = 0.973;
304 
305  if (P_tract <= 0) return alpha1;
306  return alpha + beta * v * 3.6 + delta * v * v * v * (3.6 * 3.6 * 3.6) + zeta * a * v;
307 }

309 Coord TraCIMobility::calculateHostPosition(const Coord& vehiclePos) const
310 {
311  Coord corPos;
312  if (hostPositionOffset >= 0.001) {
313  // calculate antenna position of vehicle according to antenna offset
314  corPos = vehiclePos - (heading.toCoord() * hostPositionOffset);
315  }
316  else {
317  corPos = vehiclePos;
318  }
319  return corPos;
320 }
