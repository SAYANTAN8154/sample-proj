36 void VehicleObstacleShadowing::filterSignal(Signal* signal)
37 {
38  auto senderPos = signal->getSenderPoa().pos.getPositionAt();
39  auto receiverPos = signal->getReceiverPoa().pos.getPositionAt();
40 
41  auto potentialObstacles = vehicleObstacleControl.getPotentialObstacles(signal->getSenderPoa().pos, signal->getReceiverPoa().pos, *signal);
42 
43  if (potentialObstacles.size() < 1) return;
44 
45  double senderHeight = senderPos.z;
46  double receiverHeight = receiverPos.z;
47  potentialObstacles.insert(potentialObstacles.begin(), std::make_pair(0, senderHeight));
48  potentialObstacles.emplace_back(senderPos.distance(receiverPos), receiverHeight);
49 
50  auto attenuationDB = VehicleObstacleControl::getVehicleAttenuationDZ(potentialObstacles, Signal(signal->getSpectrum()));
51 
52  EV_TRACE << "t=" << simTime() << ": Attenuation by vehicles is " << attenuationDB << std::endl;
53 
54  // convert from "dB loss" to a multiplicative factor
55  Signal attenuation(attenuationDB.getSpectrum());
56  for (uint16_t i = 0; i < attenuation.getNumValues(); i++) {
57  attenuation.at(i) = pow(10.0, -attenuationDB.at(i) / 10.0);
58  }
59 
60  *signal *= attenuation;
61 }
