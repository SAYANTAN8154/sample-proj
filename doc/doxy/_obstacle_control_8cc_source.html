36 veins::BBoxLookup rebuildBBoxLookup(const std::vector<std::unique_ptr<veins::Obstacle>>& obstacleOwner, int gridCellSize = 250)
37 {
38  std::vector<veins::Obstacle*> obstaclePointers;
39  obstaclePointers.reserve(obstacleOwner.size());
40  std::transform(obstacleOwner.begin(), obstacleOwner.end(), std::back_inserter(obstaclePointers), [](const std::unique_ptr<veins::Obstacle>& obstacle) { return obstacle.get(); });
41  auto playgroundSize = veins::FindModule<veins::BaseWorldUtility*>::findGlobalModule()->getPgs();
42  auto bboxFunction = [](veins::Obstacle* o) { return veins::BBoxLookup::Box{{o->getBboxP1().x, o->getBboxP1().y}, {o->getBboxP2().x, o->getBboxP2().y}}; };
43  return veins::BBoxLookup(obstaclePointers, bboxFunction, playgroundSize->x, playgroundSize->y, gridCellSize);
44 }

91 void ObstacleControl::addFromXml(cXMLElement* xml)
92 {
93  std::string rootTag = xml->getTagName();
94  if (rootTag != "obstacles") {
95  throw cRuntimeError("Obstacle definition root tag was \"%s\", but expected \"obstacles\"", rootTag.c_str());
96  }
97 
98  cXMLElementList list = xml->getChildren();
99  for (cXMLElementList::const_iterator i = list.begin(); i != list.end(); ++i) {
100  cXMLElement* e = *i;
101 
102  std::string tag = e->getTagName();
103 
104  if (tag == "type") {
105  // <type id="building" db-per-cut="9" db-per-meter="0.4" />
106 
107  ASSERT(e->getAttribute("id"));
108  std::string id = e->getAttribute("id");
109  ASSERT(e->getAttribute("db-per-cut"));
110  std::string perCutParS = e->getAttribute("db-per-cut");
111  double perCutPar = strtod(perCutParS.c_str(), nullptr);
112  ASSERT(e->getAttribute("db-per-meter"));
113  std::string perMeterParS = e->getAttribute("db-per-meter");
114  double perMeterPar = strtod(perMeterParS.c_str(), nullptr);
115 
116  perCut[id] = perCutPar;
117  perMeter[id] = perMeterPar;
118  }
119  else if (tag == "poly") {
120 
121  // <poly id="building#0" type="building" color="#F00" shape="16,0 8,13.8564 -8,13.8564 -16,0 -8,-13.8564 8,-13.8564" />
122  ASSERT(e->getAttribute("id"));
123  std::string id = e->getAttribute("id");
124  ASSERT(e->getAttribute("type"));
125  std::string type = e->getAttribute("type");
126  ASSERT(e->getAttribute("color"));
127  std::string color = e->getAttribute("color");
128  ASSERT(e->getAttribute("shape"));
129  std::string shape = e->getAttribute("shape");
130 
131  Obstacle obs(id, type, getAttenuationPerCut(type), getAttenuationPerMeter(type));
132  std::vector<Coord> sh;
133  cStringTokenizer st(shape.c_str());
134  while (st.hasMoreTokens()) {
135  std::string xy = st.nextToken();
136  std::vector<double> xya = cStringTokenizer(xy.c_str(), ",").asDoubleVector();
137  ASSERT(xya.size() == 2);
138  sh.push_back(Coord(xya[0], xya[1]));
139  }
140  obs.setShape(sh);
141  add(obs);
142  }
143  else {
144  throw cRuntimeError("Found unknown tag in obstacle definition: \"%s\"", tag.c_str());
145  }
146  }
147 }

186 std::vector<std::pair<veins::Obstacle*, std::vector<double>>> ObstacleControl::getIntersections(const Coord& senderPos, const Coord& receiverPos) const
187 {
188  std::vector<std::pair<Obstacle*, std::vector<double>>> allIntersections;
189 
190  // rebuild bounding box lookup structure if dirty (new obstacles added recently)
191  if (isBboxLookupDirty) {
192  bboxLookup = rebuildBBoxLookup(obstacleOwner);
193  isBboxLookupDirty = false;
194  }
195 
196  auto candidateObstacles = bboxLookup.findOverlapping({senderPos.x, senderPos.y}, {receiverPos.x, receiverPos.y});
197 
198  // remove duplicates
199  sort(candidateObstacles.begin(), candidateObstacles.end());
200  candidateObstacles.erase(unique(candidateObstacles.begin(), candidateObstacles.end()), candidateObstacles.end());
201 
202  for (Obstacle* o : candidateObstacles) {
203  // if obstacles has neither borders nor matter: bail.
204  if (o->getShape().size() < 2) continue;
205  auto foundIntersections = o->getIntersections(senderPos, receiverPos);
206  if (!foundIntersections.empty() || o->containsPoint(senderPos) || o->containsPoint(receiverPos)) {
207  allIntersections.emplace_back(o, foundIntersections);
208  }
209  }
210  return allIntersections;
211 }

213 double ObstacleControl::calculateAttenuation(const Coord& senderPos, const Coord& receiverPos) const
214 {
215  Enter_Method_Silent();
216 
217  if ((perCut.size() == 0) || (perMeter.size() == 0)) {
218  throw cRuntimeError("Unable to use SimpleObstacleShadowing: No obstacle types have been configured");
219  }
220  if (obstacleOwner.size() == 0) {
221  throw cRuntimeError("Unable to use SimpleObstacleShadowing: No obstacles have been added");
222  }
223 
224  // return cached result, if available
225  CacheKey cacheKey(senderPos, receiverPos);
226  CacheEntries::const_iterator cacheEntryIter = cacheEntries.find(cacheKey);
227  if (cacheEntryIter != cacheEntries.end()) {
228  return cacheEntryIter->second;
229  }
230 
231  // get intersections
232  auto intersections = getIntersections(senderPos, receiverPos);
233 
234  double factor = 1;
235  for (auto i = intersections.begin(); i != intersections.end(); ++i) {
236  auto o = i->first;
237  auto intersectAt = i->second;
238 
239  // if beam interacts with neither borders nor matter: bail.
240  bool senderInside = o->containsPoint(senderPos);
241  bool receiverInside = o->containsPoint(receiverPos);
242  if ((intersectAt.size() == 0) && !senderInside && !receiverInside) continue;
243 
244  // remember number of cuts before messing with intersection points
245  double numCuts = intersectAt.size();
246 
247  // for distance calculation, make sure every other pair of points marks transition through matter and void, respectively.
248  if (senderInside) intersectAt.insert(intersectAt.begin(), 0);
249  if (receiverInside) intersectAt.push_back(1);
250  ASSERT((intersectAt.size() % 2) == 0);
251 
252  // sum up distances in matter.
253  double fractionInObstacle = 0;
254  for (auto i = intersectAt.begin(); i != intersectAt.end();) {
255  double p1 = *(i++);
256  double p2 = *(i++);
257  fractionInObstacle += (p2 - p1);
258  }
259 
260  // calculate attenuation
261  double totalDistance = senderPos.distance(receiverPos);
262  double attenuation = (o->getAttenuationPerCut() * numCuts) + (o->getAttenuationPerMeter() * fractionInObstacle * totalDistance);
263  factor *= pow(10.0, -attenuation / 10.0);
264 
265  // bail if attenuation is already extremely high
266  if (factor < 1e-30) break;
267  }
268 
269  // cache result
270  if (cacheEntries.size() >= 1000) cacheEntries.clear();
271  cacheEntries[cacheKey] = factor;
272 
273  return factor;
274 }
