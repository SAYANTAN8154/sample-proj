42 Signal getMaxInterference(simtime_t start, simtime_t end, AirFrame* const referenceFrame, AirFrameVector& interfererFrames)
43 {
44  Spectrum spectrum = referenceFrame->getSignal().getSpectrum();
45  Signal maxInterference(spectrum);
46  Signal currentInterference(spectrum);
47  std::priority_queue<Signal, std::vector<Signal>, greaterByReceptionEnd<Signal>> signalEndings;
48  simtime_t currentTime = 0;
49 
50  interfererFrames.sort([](const AirFrame* x, const AirFrame* y) { return x->getConstSignal().getReceptionStart() < y->getConstSignal().getReceptionStart(); });
51 
52  for (auto& interfererFrame : interfererFrames) {
53  if (interfererFrame->getTreeId() == referenceFrame->getTreeId()) continue; // skip the signal we want to compare to
54  const Signal& signal = interfererFrame->getSignal();
55  if (signal.getReceptionEnd() <= start || signal.getReceptionStart() > end) continue; // skip signals outside our interval of interest
56  ASSERT(signal.getReceptionEnd() > start); // fail on signals ending before start of interval of interest (should be filtered out anyways)
57  ASSERT(signal.getReceptionStart() <= end); // fail on signal starting aftser the interval of interest
58  ASSERT(signal.getReceptionStart() >= currentTime); // assume frames are sorted by reception start time
59  ASSERT(signal.getSpectrum() == spectrum);
60  // fetch next signal and advance current time to its start
61  signalEndings.push(signal);
62  currentTime = signal.getReceptionStart();
63 
64  // abort at end time
65  if (currentTime >= end) break;
66 
67  // remove signals ending before the start of the current one
68  while (signalEndings.top().getReceptionEnd() <= currentTime) {
69  currentInterference -= signalEndings.top();
70  signalEndings.pop();
71  }
72 
73  // add curent signal to current total interference
74  currentInterference += signal;
75 
76  // update maximum observed interference
77  for (uint16_t spectrumIndex = signal.getDataStart(); spectrumIndex < signal.getDataEnd(); spectrumIndex++) {
78  maxInterference.at(spectrumIndex) = std::max(currentInterference.at(spectrumIndex), maxInterference.at(spectrumIndex));
79  }
80  }
81 
82  return maxInterference;
83 }

96 bool VEINS_API isChannelPowerBelowThreshold(simtime_t now, AirFrameVector& interfererFrames, size_t freqIndex, double threshold, AirFrame* exclude)
97 {
98  if (interfererFrames.empty()) {
99  // No interferers, so the channel interference is below any threshold
100  return true;
101  }
102 
103  // extract valid signals on the channel at the time of interest
104  // TODO: possibly move this filtering outside of this function
105  std::vector<Signal*> interferers;
106  for (auto& interfererFrame : interfererFrames) {
107  Signal* interferer = &interfererFrame->getSignal();
108  if (interferer->getReceptionStart() <= now && interferer->getReceptionEnd() > now && interfererFrame != exclude) {
109  interferers.push_back(interferer);
110  }
111  }
112 
113  // check once before applying analogModels
114  if (powerLevelSumAtFrequencyIndex(interferers, freqIndex) < threshold) {
115  return true;
116  }
117 
118  // start applying analogue models
119  auto analogueModelCount = interfererFrames.front()->getSignal().getAnalogueModelList()->size();
120  for (auto signalPtr : interferers) {
121  ASSERT(analogueModelCount == signalPtr->getAnalogueModelList()->size());
122  }
123  for (size_t analogueModelIndex = 0; analogueModelIndex < analogueModelCount; ++analogueModelIndex) {
124  for (auto signalPtr : interferers) {
125  signalPtr->applyAnalogueModel(analogueModelIndex);
126  }
127  if (powerLevelSumAtFrequencyIndex(interferers, freqIndex) < threshold) {
128  return true;
129  }
130  }
131 
132  // After all attenuation is performed, there interference is still higher than the threshold
133  ASSERT(powerLevelSumAtFrequencyIndex(interferers, freqIndex) >= threshold);
134  return false;
135 }

137 double VEINS_API getMinSINR(simtime_t start, simtime_t end, AirFrame* signalFrame, AirFrameVector& interfererFrames, double noise)
138 {
139  ASSERT(start >= signalFrame->getSignal().getReceptionStart());
140  ASSERT(end <= signalFrame->getSignal().getReceptionEnd());
141 
142  // Make sure all filters are applied
143  signalFrame->getSignal().applyAllAnalogueModels();
144  for (auto& interfererFrame : interfererFrames) {
145  interfererFrame->getSignal().applyAllAnalogueModels();
146  }
147 
148  Signal& signal = signalFrame->getSignal();
149  Spectrum spectrum = signal.getSpectrum();
150 
151  Signal interference = getMaxInterference(start, end, signalFrame, interfererFrames);
152  Signal sinr = signal / (interference + noise);
153 
154  double min_sinr = INFINITY;
155  for (uint16_t i = signal.getDataStart(); i < signal.getDataEnd(); i++) {
156  min_sinr = std::min(min_sinr, sinr.at(i));
157  }
158  return min_sinr;
159 }
