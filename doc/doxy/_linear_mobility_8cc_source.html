60 void LinearMobility::makeMove()
61 {
62  EV_TRACE << "start makeMove " << move.info() << endl;
63 
64  move.setStart(stepTarget, simTime());
65 
66  stepTarget.x = (move.getStartPos().x + move.getSpeed() * cos(M_PI * angle / 180) * SIMTIME_DBL(updateInterval));
67  stepTarget.y = (move.getStartPos().y + move.getSpeed() * sin(M_PI * angle / 180) * SIMTIME_DBL(updateInterval));
68 
69  move.setDirectionByTarget(stepTarget);
70 
71  EV_TRACE << "new stepTarget: " << stepTarget.info() << endl;
72 
73  // accelerate
74  move.setSpeed(move.getSpeed() + acceleration * SIMTIME_DBL(updateInterval));
75 
76  fixIfHostGetsOutside();
77 }
