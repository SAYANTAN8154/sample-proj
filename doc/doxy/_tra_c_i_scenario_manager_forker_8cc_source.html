118 int TraCIScenarioManagerForker::getPortNumber() const
119 {
120  int port = TraCIScenarioManager::getPortNumber();
121  if (port != -1) {
122  return port;
123  }
124 
125  // find a free port for the forker if port is still -1
126 
127  if (initsocketlibonce() != 0) throw cRuntimeError("Could not init socketlib");
128 
129  SOCKET sock = ::socket(AF_INET, SOCK_STREAM, 0);
130  if (sock == INVALID_SOCKET) {
131  throw cRuntimeError("Failed to create socket: %s", strerror(errno));
132  }
133 
134  struct sockaddr_in serv_addr;
135  struct sockaddr* serv_addr_p = (struct sockaddr*) &serv_addr;
136  memset(serv_addr_p, 0, sizeof(serv_addr));
137  serv_addr.sin_family = AF_INET;
138  serv_addr.sin_addr.s_addr = htonl(INADDR_ANY);
139  serv_addr.sin_port = 0;
140 
141  if (::bind(sock, serv_addr_p, sizeof(serv_addr)) < 0) {
142  throw cRuntimeError("Failed to bind socket: %s", strerror(errno));
143  }
144 
145  socklen_t len = sizeof(serv_addr);
146  if (getsockname(sock, serv_addr_p, &len) < 0) {
147  throw cRuntimeError("Failed to get hostname: %s", strerror(errno));
148  }
149 
150  port = ntohs(serv_addr.sin_port);
151 
152  closesocket(sock);
153  return port;
154 }
