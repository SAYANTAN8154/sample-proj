44 void Mac1609_4::initialize(int stage)
45 {
46  BaseMacLayer::initialize(stage);
47  if (stage == 0) {
48 
49  phy11p = FindModule<Mac80211pToPhy11pInterface*>::findSubModule(getParentModule());
50  ASSERT(phy11p);
51 
52  // this is required to circumvent double precision issues with constants from CONST80211p.h
53  ASSERT(simTime().getScaleExp() == -12);
54 
55  txPower = par("txPower").doubleValue();
56  int bitrate = par("bitrate");
57  setParametersForBitrate(bitrate);
58 
59  // unicast parameters
60  dot11RTSThreshold = par("dot11RTSThreshold");
61  dot11ShortRetryLimit = par("dot11ShortRetryLimit");
62  dot11LongRetryLimit = par("dot11LongRetryLimit");

65  frameErrorRate = par("frameErrorRate").doubleValue();
66  ackErrorRate = par("ackErrorRate").doubleValue();
67  rxStartIndication = false;
68  ignoreChannelState = false;
69  waitUntilAckRXorTimeout = false;
70  stopIgnoreChannelStateMsg = new cMessage("ChannelStateMsg");
71 
72  myId = getParentModule()->getParentModule()->getFullPath();
73  // create two edca systems
74 
75  myEDCA[ChannelType::control] = make_unique<EDCA>(this, ChannelType::control, par("queueSize"));

84  myEDCA[ChannelType::service]->myId = myId;
85  myEDCA[ChannelType::service]->myId.append(" SCH");
86  myEDCA[ChannelType::service]->createQueue(2, (((CWMIN_11P + 1) / 4) - 1), (((CWMIN_11P + 1) / 2) - 1), AC_VO);
87  myEDCA[ChannelType::service]->createQueue(3, (((CWMIN_11P + 1) / 2) - 1), CWMIN_11P, AC_VI);
88  myEDCA[ChannelType::service]->createQueue(6, CWMIN_11P, CWMAX_11P, AC_BE);

93  if (useAcks) throw cRuntimeError("Unicast model does not support channel switching");
94  // set the initial service channel

100  case 2:

103  case 3:

159  channelIdle(true);

163 void Mac1609_4::handleSelfMsg(cMessage* msg)
164 {
165  if (msg == stopIgnoreChannelStateMsg) {
166  ignoreChannelState = false;
167  return;
168  }
169 
170  if (AckTimeOutMessage* ackTimeOutMsg = dynamic_cast<AckTimeOutMessage*>(msg)) {
171  handleAckTimeOut(ackTimeOutMsg);
172  return;
173  }
174 
175  if (msg == nextChannelSwitch) {
176  ASSERT(useSCH);
177 
178  scheduleAt(simTime() + SWITCHING_INTERVAL_11P, nextChannelSwitch);
179 
180  switch (activeChannel) {
181  case ChannelType::control:
182  EV_TRACE << "CCH --> SCH" << std::endl;
183  channelBusySelf(false);
184  setActiveChannel(ChannelType::service);
185  channelIdle(true);
186  phy11p->changeListeningChannel(mySCH);
187  break;
188  case ChannelType::service:
189  EV_TRACE << "SCH --> CCH" << std::endl;
190  channelBusySelf(false);
191  setActiveChannel(ChannelType::control);
192  channelIdle(true);
193  phy11p->changeListeningChannel(Channel::cch);
194  break;
195  }
196  // schedule next channel switch in 50ms
197  }
198  else if (msg == nextMacEvent) {
199 
200  // we actually came to the point where we can send a packet
201  channelBusySelf(true);
202  BaseFrame1609_4* pktToSend = myEDCA[activeChannel]->initiateTransmit(lastIdle);
203  ASSERT(pktToSend);
204 
205  lastAC = mapUserPriority(pktToSend->getUserPriority());
206  lastWSM = pktToSend;
207 
208  EV_TRACE << "MacEvent received. Trying to send packet with priority" << lastAC << std::endl;
209 
210  // send the packet
211  Mac80211Pkt* mac = new Mac80211Pkt(pktToSend->getName(), pktToSend->getKind());
212  if (pktToSend->getRecipientAddress() != LAddress::L2BROADCAST()) {
213  mac->setDestAddr(pktToSend->getRecipientAddress());
214  }
215  else {
216  mac->setDestAddr(LAddress::L2BROADCAST());
217  }
218  mac->setSrcAddr(myMacAddr);
219  mac->encapsulate(pktToSend->dup());
220 
221  MCS usedMcs = mcs;
222  double txPower_mW;
223  PhyControlMessage* controlInfo = dynamic_cast<PhyControlMessage*>(pktToSend->getControlInfo());
224  if (controlInfo) {
225  // if MCS is not specified, just use the default one
226  MCS explicitMcs = static_cast<MCS>(controlInfo->getMcs());
227  if (explicitMcs != MCS::undefined) {
228  usedMcs = explicitMcs;
229  }
230  // apply the same principle to tx power
231  txPower_mW = controlInfo->getTxPower_mW();
232  if (txPower_mW < 0) {
233  txPower_mW = txPower;
234  }
235  }
236  else {
237  txPower_mW = txPower;
238  }
239 
240  simtime_t sendingDuration = RADIODELAY_11P + phy11p->getFrameDuration(mac->getBitLength(), usedMcs);
241  EV_TRACE << "Sending duration will be" << sendingDuration << std::endl;
242  if ((!useSCH) || (timeLeftInSlot() > sendingDuration)) {
243  if (useSCH) EV_TRACE << " Time in this slot left: " << timeLeftInSlot() << std::endl;
244 
245  Channel channelNr = (activeChannel == ChannelType::control) ? Channel::cch : mySCH;
246  double freq = IEEE80211ChannelFrequencies.at(channelNr);
247 
248  EV_TRACE << "Sending a Packet. Frequency " << freq << " Priority" << lastAC << std::endl;
249  sendFrame(mac, RADIODELAY_11P, channelNr, usedMcs, txPower_mW);
250 
251  // schedule ack timeout for unicast packets
252  if (pktToSend->getRecipientAddress() != LAddress::L2BROADCAST() && useAcks) {
253  waitUntilAckRXorTimeout = true;
254  // PHY-RXSTART.indication should be received within ackWaitTime
255  // sifs + slot + rx_delay: see 802.11-2012 9.3.2.8 (32us + 13us + 49us = 94us)
256  simtime_t ackWaitTime(94, SIMTIME_US);
257  // update id in the retransmit timer
258  myEDCA[activeChannel]->myQueues[lastAC].ackTimeOut->setWsmId(pktToSend->getTreeId());
259  simtime_t timeOut = sendingDuration + ackWaitTime;
260  scheduleAt(simTime() + timeOut, myEDCA[activeChannel]->myQueues[lastAC].ackTimeOut);
261  }
262  }
263  else { // not enough time left now
264  EV_TRACE << "Too little Time left. This packet cannot be send in this slot." << std::endl;
265  statsNumTooLittleTime++;
266  // revoke TXOP
267  myEDCA[activeChannel]->revokeTxOPs();
268  delete mac;
269  channelIdle();
270  // do nothing. contention will automatically start after channel switch
271  }
272  }

280 void Mac1609_4::handleUpperMsg(cMessage* msg)
281 {
282 
283  BaseFrame1609_4* thisMsg = check_and_cast<BaseFrame1609_4*>(msg);
284 
285  t_access_category ac = mapUserPriority(thisMsg->getUserPriority());
286 
287  EV_TRACE << "Received a message from upper layer for channel " << thisMsg->getChannelNumber() << " Access Category (Priority): " << ac << std::endl;
288 
289  ChannelType chan;
290 
291  if (static_cast<Channel>(thisMsg->getChannelNumber()) == Channel::cch) {
292  chan = ChannelType::control;
293  }
294  else {
295  ASSERT(useSCH);
296  thisMsg->setChannelNumber(static_cast<int>(mySCH));
297  chan = ChannelType::service;
298  }
299 
300  int num = myEDCA[chan]->queuePacket(ac, thisMsg);
301 
302  // packet was dropped in Mac
303  if (num == -1) {
304  statsDroppedPackets++;
305  return;
306  }
307 
308  // if this packet is not at the front of a new queue we dont have to reevaluate times
309  EV_TRACE << "sorted packet into queue of EDCA " << static_cast<int>(chan) << " this packet is now at position: " << num << std::endl;
310 
311  if (chan == activeChannel) {
312  EV_TRACE << "this packet is for the currently active channel" << std::endl;
313  }
314  else {
315  EV_TRACE << "this packet is NOT for the currently active channel" << std::endl;
316  }
317 
318  if (num == 1 && idleChannel == true && chan == activeChannel) {
319 
320  simtime_t nextEvent = myEDCA[chan]->startContent(lastIdle, guardActive());
321 
322  if (nextEvent != -1) {
323  if ((!useSCH) || (nextEvent <= nextChannelSwitch->getArrivalTime())) {
324  if (nextMacEvent->isScheduled()) {
325  cancelEvent(nextMacEvent);
326  }
327  scheduleAt(nextEvent, nextMacEvent);
328  EV_TRACE << "Updated nextMacEvent:" << nextMacEvent->getArrivalTime().raw() << std::endl;
329  }
330  else {
331  EV_TRACE << "Too little time in this interval. Will not schedule nextMacEvent" << std::endl;
332  // it is possible that this queue has an txop. we have to revoke it
333  myEDCA[activeChannel]->revokeTxOPs();
334  statsNumTooLittleTime++;
335  }
336  }
337  else {
338  cancelEvent(nextMacEvent);
339  }
340  }
341  if (num == 1 && idleChannel == false && myEDCA[chan]->myQueues[ac].currentBackoff == 0 && chan == activeChannel) {
342  myEDCA[chan]->backoff(ac);
343  }
344 }

346 void Mac1609_4::handleLowerControl(cMessage* msg)
347 {
348  if (msg->getKind() == MacToPhyInterface::PHY_RX_START) {
349  rxStartIndication = true;
350  }
351  else if (msg->getKind() == MacToPhyInterface::PHY_RX_END_WITH_SUCCESS) {
352  // PHY_RX_END_WITH_SUCCESS will get packet soon! Nothing to do here
353  }
354  else if (msg->getKind() == MacToPhyInterface::PHY_RX_END_WITH_FAILURE) {
355  // RX failed at phy. Time to retransmit
356  phy11p->notifyMacAboutRxStart(false);
357  rxStartIndication = false;
358  handleRetransmit(lastAC);
359  }
360  else if (msg->getKind() == MacToPhyInterface::TX_OVER) {
361 
362  EV_TRACE << "Successfully transmitted a packet on " << lastAC << std::endl;
363 
364  phy->setRadioState(Radio::RX);
365 
366  if (!dynamic_cast<Mac80211Ack*>(lastMac.get())) {
367  // message was sent
368  // update EDCA queue. go into post-transmit backoff and set cwCur to cwMin
369  myEDCA[activeChannel]->postTransmit(lastAC, lastWSM, useAcks);
370  }
371  // channel just turned idle.
372  // don't set the chan to idle. the PHY layer decides, not us.
373 
374  if (guardActive()) {
375  throw cRuntimeError("We shouldnt have sent a packet in guard!");
376  }
377  }
378  else if (msg->getKind() == Mac80211pToPhy11pInterface::CHANNEL_BUSY) {
379  channelBusy();
380  }
381  else if (msg->getKind() == Mac80211pToPhy11pInterface::CHANNEL_IDLE) {
382  // Decider80211p::processSignalEnd() sends up the received packet to MAC followed by control message CHANNEL_IDLE in the same timestamp.
383  // If we received a unicast frame (first event scheduled by Decider), MAC immediately schedules an ACK message and wants to switch the radio to TX mode.
384  // So, the notification for channel idle from phy is undesirable and we skip it here.
385  // After ACK TX is over, PHY will inform the channel status again.
386  if (ignoreChannelState) {
387  // Skipping channelidle because we are about to send an ack regardless of the channel state
388  }
389  else {
390  channelIdle();
391  }
392  }
393  else if (msg->getKind() == Decider80211p::BITERROR || msg->getKind() == Decider80211p::COLLISION) {
394  statsSNIRLostPackets++;
395  EV_TRACE << "A packet was not received due to biterrors" << std::endl;
396  }
397  else if (msg->getKind() == Decider80211p::RECWHILESEND) {
398  statsTXRXLostPackets++;
399  EV_TRACE << "A packet was not received because we were sending while receiving" << std::endl;
400  }
401  else if (msg->getKind() == MacToPhyInterface::RADIO_SWITCHING_OVER) {
402  EV_TRACE << "Phylayer said radio switching is done" << std::endl;
403  }
404  else if (msg->getKind() == BaseDecider::PACKET_DROPPED) {
405  phy->setRadioState(Radio::RX);
406  EV_TRACE << "Phylayer said packet was dropped" << std::endl;
407  }
408  else {
409  EV_WARN << "Invalid control message type (type=NOTHING) : name=" << msg->getName() << " modulesrc=" << msg->getSenderModule()->getFullPath() << "." << std::endl;
410  ASSERT(false);
411  }
412 
413  if (msg->getKind() == Decider80211p::COLLISION) {
414  emit(sigCollision, true);
415  }
416 
417  delete msg;
418 }

468 void Mac1609_4::sendFrame(Mac80211Pkt* frame, simtime_t delay, Channel channelNr, MCS mcs, double txPower_mW)
469 {
470  phy->setRadioState(Radio::TX); // give time for the radio to be in Tx state before transmitting
471 
472  delay = std::max(delay, RADIODELAY_11P); // wait at least for the radio to switch
473 
474  attachControlInfo(frame, channelNr, mcs, txPower_mW);
475  check_and_cast<MacToPhyControlInfo11p*>(frame->getControlInfo());
476 
477  lastMac.reset(frame->dup());
478  sendDelayed(frame, delay, lowerLayerOut);
479 
480  if (dynamic_cast<Mac80211Ack*>(frame)) {
481  statsSentAcks += 1;
482  emit(sigSentAck, true);
483  }
484  else {
485  statsSentPackets += 1;
486  emit(sigSentPacket, true);
487  }

564 void Mac1609_4::handleLowerMsg(cMessage* msg)
565 {
566  Mac80211Pkt* macPkt = check_and_cast<Mac80211Pkt*>(msg);
567 
568  // pass information about received frame to the upper layers
569  DeciderResult80211* macRes = check_and_cast<DeciderResult80211*>(PhyToMacControlInfo::getDeciderResult(msg));
570  DeciderResult80211* res = new DeciderResult80211(*macRes);
571 
572  long dest = macPkt->getDestAddr();
573 
574  EV_TRACE << "Received frame name= " << macPkt->getName() << ", myState= src=" << macPkt->getSrcAddr() << " dst=" << macPkt->getDestAddr() << " myAddr=" << myMacAddr << std::endl;
575 
576  bool frameReceived = true;
577  if (dblrand() < frameErrorRate) frameReceived = false;
578 
579  if (dest == myMacAddr) {
580  if (auto* ack = dynamic_cast<Mac80211Ack*>(macPkt)) {
581  ASSERT(useAcks);
582  if (dblrand() >= ackErrorRate)
583  handleAck(ack);
584  else
585  EV_TRACE << "Artificially dropping ACK";
586  delete res;
587  }
588  else {
589  if (frameReceived) {
590  unique_ptr<BaseFrame1609_4> wsm(check_and_cast<BaseFrame1609_4*>(macPkt->decapsulate()));
591  auto ctrlInfo = new PhyToMacControlInfo(res);
592  ctrlInfo->setSourceAddress(macPkt->getSrcAddr());
593  wsm->setControlInfo(ctrlInfo);

675 BaseFrame1609_4* Mac1609_4::EDCA::initiateTransmit(simtime_t lastIdle)
676 {
677 
678  // iterate through the queues to return the packet we want to send
679  BaseFrame1609_4* pktToSend = nullptr;
680 
681  simtime_t idleTime = simTime() - lastIdle;
682 
683  EV_TRACE << "Initiating transmit at " << simTime() << ". I've been idle since " << idleTime << std::endl;
684 
685  // As t_access_category is sorted by priority, we iterate back to front.
686  // This realizes the behavior documented in IEEE Std 802.11-2012 Section 9.2.4.2; that is, "data frames from the higher priority AC" win an internal collision.
687  // The phrase "EDCAF of higher UP" of IEEE Std 802.11-2012 Section 9.19.2.3 is assumed to be meaningless.
688  for (auto iter = myQueues.rbegin(); iter != myQueues.rend(); iter++) {
689  if (iter->second.queue.size() != 0 && !iter->second.waitForAck) {
690  if (idleTime >= iter->second.aifsn * SLOTLENGTH_11P + SIFS_11P && iter->second.txOP == true) {
691 
692  EV_TRACE << "Queue " << iter->first << " is ready to send!" << std::endl;
693 
694  iter->second.txOP = false;
695  // this queue is ready to send
696  if (pktToSend == nullptr) {
697  pktToSend = iter->second.queue.front();
698  }
699  else {
700  // there was already another packet ready. we have to go increase cw and go into backoff. It's called internal contention and its wonderful
701 
702  statsNumInternalContention++;
703  iter->second.cwCur = std::min(iter->second.cwMax, (iter->second.cwCur + 1) * 2 - 1);
704  iter->second.currentBackoff = owner->intuniform(0, iter->second.cwCur);
705  EV_TRACE << "Internal contention for queue " << iter->first << " : " << iter->second.currentBackoff << ". Increase cwCur to " << iter->second.cwCur << std::endl;
706  }
707  }
708  }
709  }
710 
711  if (pktToSend == nullptr) {
712  throw cRuntimeError("No packet was ready");
713  }
714  return pktToSend;
715 }

717 simtime_t Mac1609_4::EDCA::startContent(simtime_t idleSince, bool guardActive)
718 {
719 
720  EV_TRACE << "Restarting contention." << std::endl;
721 
722  simtime_t nextEvent = -1;
723 
724  simtime_t idleTime = SimTime().setRaw(std::max((int64_t) 0, (simTime() - idleSince).raw()));
725  ;
726 
727  lastStart = idleSince;
728 
729  EV_TRACE << "Channel is already idle for:" << idleTime << " since " << idleSince << std::endl;
730 
731  // this returns the nearest possible event in this EDCA subsystem after a busy channel
732 
733  for (auto&& p : myQueues) {
734  auto& accessCategory = p.first;
735  auto& edcaQueue = p.second;
736  if (edcaQueue.queue.size() != 0 && !edcaQueue.waitForAck) {
737 
738  /* 1609_4 says that when attempting to send (backoff == 0) when guard is active, a random backoff is invoked */
739 
740  if (guardActive == true && edcaQueue.currentBackoff == 0) {
741  // cw is not increased
742  edcaQueue.currentBackoff = owner->intuniform(0, edcaQueue.cwCur);
743  statsNumBackoff++;
744  }
745 
746  simtime_t DIFS = edcaQueue.aifsn * SLOTLENGTH_11P + SIFS_11P;
747 
748  // the next possible time to send can be in the past if the channel was idle for a long time, meaning we COULD have sent earlier if we had a packet
749  simtime_t possibleNextEvent = DIFS + edcaQueue.currentBackoff * SLOTLENGTH_11P;
750 
751  EV_TRACE << "Waiting Time for Queue " << accessCategory << ":" << possibleNextEvent << "=" << edcaQueue.aifsn << " * " << SLOTLENGTH_11P << " + " << SIFS_11P << "+" << edcaQueue.currentBackoff << "*" << SLOTLENGTH_11P << "; Idle time: " << idleTime << std::endl;
752 
753  if (idleTime > possibleNextEvent) {
754  EV_TRACE << "Could have already send if we had it earlier" << std::endl;
755  // we could have already sent. round up to next boundary
756  simtime_t base = idleSince + DIFS;
757  possibleNextEvent = simTime() - simtime_t().setRaw((simTime() - base).raw() % SLOTLENGTH_11P.raw()) + SLOTLENGTH_11P;
758  }
759  else {
760  // we are gonna send in the future
761  EV_TRACE << "Sending in the future" << std::endl;
762  possibleNextEvent = idleSince + possibleNextEvent;
763  }
764  nextEvent == -1 ? nextEvent = possibleNextEvent : nextEvent = std::min(nextEvent, possibleNextEvent);
765  }
766  }
767  return nextEvent;

770 void Mac1609_4::EDCA::stopContent(bool allowBackoff, bool generateTxOp)
771 {
772  // update all Queues
773 
774  EV_TRACE << "Stopping Contention at " << simTime().raw() << std::endl;
775 
776  simtime_t passedTime = simTime() - lastStart;
777 
778  EV_TRACE << "Channel was idle for " << passedTime << std::endl;
779 
780  lastStart = -1; // indicate that there was no last start
781 
782  for (auto&& p : myQueues) {
783  auto& accessCategory = p.first;
784  auto& edcaQueue = p.second;
785  if ((edcaQueue.currentBackoff != 0 || edcaQueue.queue.size() != 0) && !edcaQueue.waitForAck) {
786  // check how many slots we already waited until the chan became busy
787 
788  int64_t oldBackoff = edcaQueue.currentBackoff;
789 
790  std::string info;
791  if (passedTime < edcaQueue.aifsn * SLOTLENGTH_11P + SIFS_11P) {
792  // we didnt even make it one DIFS :(
793  info.append(" No DIFS");
794  }
795  else {
796  // decrease the backoff by one because we made it longer than one DIFS
797  edcaQueue.currentBackoff -= 1;
798 
799  // check how many slots we waited after the first DIFS
800  int64_t passedSlots = (int64_t)((passedTime - SimTime(edcaQueue.aifsn * SLOTLENGTH_11P + SIFS_11P)) / SLOTLENGTH_11P);
801 
802  EV_TRACE << "Passed slots after DIFS: " << passedSlots << std::endl;
803 
804  if (edcaQueue.queue.size() == 0) {
805  // this can be below 0 because of post transmit backoff -> backoff on empty queues will not generate macevents,
806  // we dont want to generate a txOP for empty queues
807  edcaQueue.currentBackoff -= std::min(edcaQueue.currentBackoff, passedSlots);
808  info.append(" PostCommit Over");
809  }
810  else {
811  edcaQueue.currentBackoff -= passedSlots;
812  if (edcaQueue.currentBackoff <= -1) {
813  if (generateTxOp) {
814  edcaQueue.txOP = true;
815  info.append(" TXOP");
816  }
817  // else: this packet couldnt be sent because there was too little time. we could have generated a txop, but the channel switched
818  edcaQueue.currentBackoff = 0;
819  }
820  }
821  }
822  EV_TRACE << "Updating backoff for Queue " << accessCategory << ": " << oldBackoff << " -> " << edcaQueue.currentBackoff << info << std::endl;
823  }
824  }

1106 void Mac1609_4::handleRetransmit(t_access_category ac)
1107 {
1108  // cancel the acktime out
1109  if (myEDCA[ChannelType::control]->myQueues[ac].ackTimeOut->isScheduled()) {
1110  // This case is possible if we received PHY_RX_END_WITH_SUCCESS or FAILURE even before ack timeout
1111  cancelEvent(myEDCA[ChannelType::control]->myQueues[ac].ackTimeOut);
1112  }
1113  if (myEDCA[ChannelType::control]->myQueues[ac].queue.size() == 0) {
1114  throw cRuntimeError("Trying retransmission on empty queue...");
1115  }
1116  BaseFrame1609_4* appPkt = myEDCA[ChannelType::control]->myQueues[ac].queue.front();
1117  bool contend = false;
1118  bool retriesExceeded = false;
1119  // page 879 of IEEE 802.11-2012
1120  if (appPkt->getBitLength() <= dot11RTSThreshold) {
1121  myEDCA[ChannelType::control]->myQueues[ac].ssrc++;
1122  if (myEDCA[ChannelType::control]->myQueues[ac].ssrc <= dot11ShortRetryLimit) {
1123  retriesExceeded = false;
1124  }
1125  else {
1126  retriesExceeded = true;
1127  }
1128  }
1129  else {
1130  myEDCA[ChannelType::control]->myQueues[ac].slrc++;
1131  if (myEDCA[ChannelType::control]->myQueues[ac].slrc <= dot11LongRetryLimit) {
1132  retriesExceeded = false;
1133  }
1134  else {
1135  retriesExceeded = true;
1136  }
1137  }
1138  if (!retriesExceeded) {
1139  // try again!
1140  myEDCA[ChannelType::control]->myQueues[ac].cwCur = std::min(myEDCA[ChannelType::control]->myQueues[ac].cwMax, (myEDCA[ChannelType::control]->myQueues[ac].cwCur * 2) + 1);
1141  myEDCA[ChannelType::control]->backoff(ac);
1142  contend = true;
1143  // no need to reset wait on id here as we are still retransmitting same packet
1144  myEDCA[ChannelType::control]->myQueues[ac].waitForAck = false;
1145  }
1146  else {
1147  // enough tries!
1148  myEDCA[ChannelType::control]->myQueues[ac].queue.pop();
1149  if (myEDCA[ChannelType::control]->myQueues[ac].queue.size() > 0) {
1150  // start contention only if there are more packets in the queue
1151  contend = true;
1152  }
1153  // notify interested applications of a failed transmission, together with the actual packet
1154  emit(sigRetriesExceeded, appPkt);
1155  statsRetriesExceeded++;
1156  delete appPkt;
1157  myEDCA[ChannelType::control]->myQueues[ac].cwCur = myEDCA[ChannelType::control]->myQueues[ac].cwMin;
1158  myEDCA[ChannelType::control]->backoff(ac);
1159  myEDCA[ChannelType::control]->myQueues[ac].waitForAck = false;
1160  myEDCA[ChannelType::control]->myQueues[ac].waitOnUnicastID = -1;
1161  myEDCA[ChannelType::control]->myQueues[ac].ssrc = 0;
1162  myEDCA[ChannelType::control]->myQueues[ac].slrc = 0;
1163  }
1164  waitUntilAckRXorTimeout = false;
1165  if (contend && idleChannel && !ignoreChannelState) {
1166  // reevaluate times -- if channel is not idle, then contention would start automatically
1167  cancelEvent(nextMacEvent);
1168  simtime_t nextEvent = myEDCA[ChannelType::control]->startContent(lastIdle, guardActive());
1169  scheduleAt(nextEvent, nextMacEvent);
1170  }
1171 }
