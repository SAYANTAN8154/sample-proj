55 simtime_t BaseDecider::processNewSignal(AirFrame* frame)
56 {
57  if (currentSignal.first != 0) {
58  EV_TRACE << "Already receiving another AirFrame!" << endl;
59  return notAgain;
60  }
61 
62  // get the receiving power of the Signal at start-time
63  Signal& signal = frame->getSignal();
64  double recvPower = signal.getMax();
65 
66  // check whether signal is strong enough to receive
67  if (recvPower < minPowerLevel) {
68  EV_TRACE << "Signal is too weak (" << recvPower << " < " << minPowerLevel << ") -> do not receive." << endl;
69  // Signal too weak, we can't receive it, tell PhyLayer that we don't want it again
70  return notAgain;
71  }
72 
73  // Signal is strong enough, receive this Signal and schedule it
74  EV_TRACE << "Signal is strong enough (" << recvPower << " > " << minPowerLevel << ") -> Trying to receive AirFrame." << endl;
75 
76  currentSignal.first = frame;
77  currentSignal.second = EXPECT_END;
78 
79  // channel turned busy
80  setChannelIdleStatus(false);
81 
82  return signal.getReceptionEnd();
83 }

85 simtime_t BaseDecider::processSignalEnd(AirFrame* frame)
86 {
87  EV_INFO << "packet was received correctly, it is now handed to upper layer...\n";
88  phy->sendUp(frame, new DeciderResult(true));
89 
90  // we have processed this AirFrame and we prepare to receive the next one
91  currentSignal.first = 0;
92 
93  // channel is idle now
94  setChannelIdleStatus(true);
95 
96  return notAgain;
97 }
