80 simtime_t Radio::switchTo(int newState, simtime_t_cref now)
81 {
82  // state to switch to must be in a valid range, i.e. 0 <= newState < numRadioStates
83  ASSERT(0 <= newState && newState < numRadioStates);
84 
85  // state to switch to must not be SWITCHING
86  ASSERT(newState != SWITCHING);
87 
88  // return error value if newState is the same as the current state
89  // if (newState == state) return -1;
90 
91  // return error value if Radio is currently switching
92  if (state == SWITCHING) return -1;
93 
94  /* REGULAR CASE */
95 
96  // set the nextState to the newState and the current state to SWITCHING
97  nextState = newState;
98  int lastState = state;
99  state = SWITCHING;
100  radioStates.record(state);
101 
102  // return matching entry from the switch times matrix
103  return swTimes[lastState][nextState];
104 }
