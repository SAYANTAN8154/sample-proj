81 std::vector<std::string> getMapping(std::string el)
82 {
83 
84  // search for string protection characters '
85  char protection = '\'';
86  size_t first = el.find(protection);
87  size_t second;
88  size_t eq;
89  std::string type, value;
90  std::vector<std::string> mapping;
91 
92  if (first == std::string::npos) {
93  // there's no string protection, simply split by '='
94  cStringTokenizer stk(el.c_str(), "=");
95  mapping = stk.asVector();
96  }
97  else {
98  // if there's string protection, we need to find a matching delimiter
99  second = el.find(protection, first + 1);
100  // ensure that a matching delimiter exists, and that it is at the end
101  if (second == std::string::npos || second != el.size() - 1) throw cRuntimeError("invalid syntax for mapping \"%s\"", el.c_str());
102 
103  // take the value of the mapping as the text within the quotes
104  value = el.substr(first + 1, second - first - 1);
105 
106  if (first == 0) {
107  // if the string starts with a quote, there's only the value
108  mapping.push_back(value);
109  }
110  else {
111  // search for the equal sign
112  eq = el.find('=');
113  // this must be the character before the quote
114  if (eq == std::string::npos || eq != first - 1) {
115  throw cRuntimeError("invalid syntax for mapping \"%s\"", el.c_str());
116  }
117  else {
118  type = el.substr(0, eq);
119  }
120  mapping.push_back(type);
121  mapping.push_back(value);
122  }
123  }
124  return mapping;
125 }

129 TraCIScenarioManager::TypeMapping TraCIScenarioManager::parseMappings(std::string parameter, std::string parameterName, bool allowEmpty)
130 {
131 
132  /**
133  * possible syntaxes
134  *
135  * "a" : assign module type "a" to all nodes (for backward compatibility)
136  * "a=b" : assign module type "b" to vehicle type "a". the presence of any other vehicle type in the simulation will cause the simulation to stop
137  * "a=b c=d" : assign module type "b" to vehicle type "a" and "d" to "c". the presence of any other vehicle type in the simulation will cause the simulation to stop
138  * "a=b c=d *=e": everything which is not of vehicle type "a" or "b", assign module type "e"
139  * "a=b c=0" : for vehicle type "c" no module should be instantiated
140  * "a=b c=d *=0": everything which is not of vehicle type a or c should not be instantiated
141  *
142  * For display strings key-value pairs needs to be protected with single quotes, as they use an = sign as the type mappings. For example
143  * *.manager.moduleDisplayString = "'i=block/process'"
144  * *.manager.moduleDisplayString = "a='i=block/process' b='i=misc/sun'"
145  *
146  * moduleDisplayString can also be left empty:
147  * *.manager.moduleDisplayString = ""
148  */
149 
150  unsigned int i;
151  TypeMapping map;
152 
153  // tokenizer to split into mappings ("a=b c=d", -> ["a=b", "c=d"])
154  cStringTokenizer typesTz(parameter.c_str(), " ");
155  // get all mappings
156  std::vector<std::string> typeMappings = typesTz.asVector();
157  // and check that there exists at least one
158  if (typeMappings.size() == 0) {
159  if (!allowEmpty)
160  throw cRuntimeError("parameter \"%s\" is empty", parameterName.c_str());
161  else
162  return map;
163  }
164 
165  // loop through all mappings
166  for (i = 0; i < typeMappings.size(); i++) {
167 
168  // tokenizer to find the mapping from vehicle type to module type
169  std::string typeMapping = typeMappings[i];
170 
171  std::vector<std::string> mapping = getMapping(typeMapping);
172 
173  if (mapping.size() == 1) {
174  // we are where there is no actual assignment
175  // "a": this is good
176  // "a b=c": this is not
177  if (typeMappings.size() != 1)
178  // stop simulation with an error
179  throw cRuntimeError("parameter \"%s\" includes multiple mappings, but \"%s\" is not mapped to any vehicle type", parameterName.c_str(), mapping[0].c_str());
180  else
181  // all vehicle types should be instantiated with this module type
182  map["*"] = mapping[0];
183  }
184  else {
185 
186  // check that mapping is valid (a=b and not like a=b=c)
187  if (mapping.size() != 2) throw cRuntimeError("invalid syntax for mapping \"%s\" for parameter \"%s\"", typeMapping.c_str(), parameterName.c_str());
188  // check that the mapping does not already exist
189  if (map.find(mapping[0]) != map.end()) throw cRuntimeError("duplicated mapping for vehicle type \"%s\" for parameter \"%s\"", mapping[0].c_str(), parameterName.c_str());
190 
191  // finally save the mapping
192  map[mapping[0]] = mapping[1];
193  }
194  }
195 
196  return map;
197 }

199 void TraCIScenarioManager::parseModuleTypes()
200 {
201 
202  TypeMapping::iterator i;
203  std::vector<std::string> typeKeys, nameKeys, displayStringKeys;
204 
205  std::string moduleTypes = par("moduleType").stdstringValue();
206  std::string moduleNames = par("moduleName").stdstringValue();
207  std::string moduleDisplayStrings = par("moduleDisplayString").stdstringValue();
208 
209  moduleType = parseMappings(moduleTypes, "moduleType", false);
210  moduleName = parseMappings(moduleNames, "moduleName", false);
211  moduleDisplayString = parseMappings(moduleDisplayStrings, "moduleDisplayString", true);
212 
213  // perform consistency check. for each vehicle type in moduleType there must be a vehicle type
214  // in moduleName (and in moduleDisplayString if moduleDisplayString is not empty)
215 
216  // get all the keys
217  for (i = moduleType.begin(); i != moduleType.end(); i++) typeKeys.push_back(i->first);
218  for (i = moduleName.begin(); i != moduleName.end(); i++) nameKeys.push_back(i->first);
219  for (i = moduleDisplayString.begin(); i != moduleDisplayString.end(); i++) displayStringKeys.push_back(i->first);
220 
221  // sort them (needed for intersection)
222  std::sort(typeKeys.begin(), typeKeys.end());
223  std::sort(nameKeys.begin(), nameKeys.end());
224  std::sort(displayStringKeys.begin(), displayStringKeys.end());
225 
226  std::vector<std::string> intersection;
227 
228  // perform set intersection
229  std::set_intersection(typeKeys.begin(), typeKeys.end(), nameKeys.begin(), nameKeys.end(), std::back_inserter(intersection));
230  if (intersection.size() != typeKeys.size() || intersection.size() != nameKeys.size()) throw cRuntimeError("keys of mappings of moduleType and moduleName are not the same");
231 
232  if (displayStringKeys.size() == 0) return;
233 
234  intersection.clear();
235  std::set_intersection(typeKeys.begin(), typeKeys.end(), displayStringKeys.begin(), displayStringKeys.end(), std::back_inserter(intersection));
236  if (intersection.size() != displayStringKeys.size()) throw cRuntimeError("keys of mappings of moduleType and moduleDisplayString are not the same");
237 }

296 void TraCIScenarioManager::init_traci()
297 {
298  auto* commandInterface = getCommandInterface();
299  {
300  auto apiVersion = commandInterface->getVersion();
301  EV_DEBUG << "TraCI server \"" << apiVersion.second << "\" reports API version " << apiVersion.first << endl;
302  commandInterface->setApiVersion(apiVersion.first);
303  }
304 
305  {
306  // query and set road network boundaries
307  auto networkBoundaries = commandInterface->initNetworkBoundaries(par("margin"));
308  if (world != nullptr && ((connection->traci2omnet(networkBoundaries.second).x > world->getPgs()->x) || (connection->traci2omnet(networkBoundaries.first).y > world->getPgs()->y))) {
309  EV_WARN << "WARNING: Playground size (" << world->getPgs()->x << ", " << world->getPgs()->y << ") might be too small for vehicle at network bounds (" << connection->traci2omnet(networkBoundaries.second).x << ", " << connection->traci2omnet(networkBoundaries.first).y << ")" << endl;
310  }
311  }
312 
313  {
314  // subscribe to list of departed and arrived vehicles, as well as simulation time
315  simtime_t beginTime = 0;
316  simtime_t endTime = SimTime::getMaxTime();
317  std::string objectId = "";
318  std::list<uint8_t> variables;
319  variables.push_back(VAR_DEPARTED_VEHICLES_IDS);
320  variables.push_back(VAR_ARRIVED_VEHICLES_IDS);
321  variables.push_back(commandInterface->getTimeStepCmd());
322  if (commandInterface->getApiVersion() >= 18) {
323  variables.push_back(VAR_COLLIDING_VEHICLES_IDS);
324  }
325  variables.push_back(VAR_TELEPORT_STARTING_VEHICLES_IDS);
326  variables.push_back(VAR_TELEPORT_ENDING_VEHICLES_IDS);
327  variables.push_back(VAR_PARKING_STARTING_VEHICLES_IDS);
328  variables.push_back(VAR_PARKING_ENDING_VEHICLES_IDS);
329  uint8_t variableNumber = variables.size();
330  TraCIBuffer buf1 = TraCIBuffer();
331  buf1 << beginTime << endTime << objectId << variableNumber;
332  for (auto variable : variables) {
333  buf1 << variable;
334  }
335  TraCIBuffer buf = connection->query(CMD_SUBSCRIBE_SIM_VARIABLE, buf1);
336 
337  processSubcriptionResult(buf);
338  ASSERT(buf.eof());
339  }
340 
341  {
342  // subscribe to list of vehicle ids
343  simtime_t beginTime = 0;
344  simtime_t endTime = SimTime::getMaxTime();
345  std::string objectId = "";
346  uint8_t variableNumber = 1;
347  uint8_t variable1 = ID_LIST;
348  TraCIBuffer buf = connection->query(CMD_SUBSCRIBE_VEHICLE_VARIABLE, TraCIBuffer() << beginTime << endTime << objectId << variableNumber << variable1);
349  processSubcriptionResult(buf);
350  ASSERT(buf.eof());
351  }
352 
353  if (!trafficLightModuleType.empty() && !trafficLightModuleIds.empty()) {
354  // initialize traffic lights
355  cModule* parentmod = getParentModule();
356  if (!parentmod) {
357  throw cRuntimeError("Parent Module not found (for traffic light creation)");
358  }
359  cModuleType* tlModuleType = cModuleType::get(trafficLightModuleType.c_str());
360 
361  // query traffic lights via TraCI
362  std::list<std::string> trafficLightIds = commandInterface->getTrafficlightIds();
363 #if OMNETPP_BUILDNUM >= 1525
364 #else
365  size_t nrOfTrafficLights = trafficLightIds.size();
366 #endif
367  int cnt = 0;
368  for (std::list<std::string>::iterator i = trafficLightIds.begin(); i != trafficLightIds.end(); ++i) {
369  std::string tlId = *i;
370  if (std::find(trafficLightModuleIds.begin(), trafficLightModuleIds.end(), tlId) == trafficLightModuleIds.end()) {
371  continue; // filter only selected elements
372  }
373 
374  Coord position = commandInterface->junction(tlId).getPosition();
375 
376 #if OMNETPP_BUILDNUM >= 1525
377  parentmod->setSubmoduleVectorSize(trafficLightModuleName.c_str(), cnt + 1);
378  cModule* module = tlModuleType->create(trafficLightModuleName.c_str(), parentmod, cnt);
379 #else
380  cModule* module = tlModuleType->create(trafficLightModuleName.c_str(), parentmod, nrOfTrafficLights, cnt);
381 #endif
382  module->par("externalId") = tlId;
383  module->finalizeParameters();
384  module->getDisplayString().parse(trafficLightModuleDisplayString.c_str());
385  module->buildInside();
386  module->scheduleStart(simTime() + updateInterval);
387 
388  cModule* tlIfSubmodule = module->getSubmodule("tlInterface");
389  // initialize traffic light interface with current program
390  TraCITrafficLightInterface* tlIfModule = dynamic_cast<TraCITrafficLightInterface*>(tlIfSubmodule);
391  tlIfModule->preInitialize(tlId, position, updateInterval);
392 
393  // initialize mobility for positioning
394  BaseMobility* mobiSubmodule = check_and_cast<BaseMobility*>(module->getSubmodule("mobility"));
395  mobiSubmodule->setStartPosition(position);
396 
397  module->callInitialize();
398  trafficLights[tlId] = module;
399  subscribeToTrafficLightVariables(tlId); // subscribe after module is in trafficLights
400  cnt++;
401  }
402  }
403 
404  ObstacleControl* obstacles = ObstacleControlAccess().getIfExists();
405  if (obstacles) {
406  {
407  // get list of polygons
408  std::list<std::string> ids = commandInterface->getPolygonIds();
409  for (std::list<std::string>::iterator i = ids.begin(); i != ids.end(); ++i) {
410  std::string id = *i;
411  std::string typeId = commandInterface->polygon(id).getTypeId();
412  if (!obstacles->isTypeSupported(typeId)) continue;
413  std::list<Coord> coords = commandInterface->polygon(id).getShape();
414  std::vector<Coord> shape;
415  std::copy(coords.begin(), coords.end(), std::back_inserter(shape));
416  for (auto p : shape) {
417  if ((p.x < 0) || (p.y < 0) || (p.x > world->getPgs()->x) || (p.y > world->getPgs()->y)) {
418  EV_WARN << "WARNING: Playground (" << world->getPgs()->x << ", " << world->getPgs()->y << ") will not fit radio obstacle at (" << p.x << ", " << p.y << ")" << endl;
419  }
420  }
421  obstacles->addFromTypeAndShape(id, typeId, shape);
422  }
423  }
424  }
425 
426  traciInitialized = true;
427  emit(traciInitializedSignal, true);
428 
429  // draw and calculate area of rois
430  for (std::list<std::pair<TraCICoord, TraCICoord>>::const_iterator r = roi.getRectangles().begin(), end = roi.getRectangles().end(); r != end; ++r) {
431  TraCICoord first = r->first;
432  TraCICoord second = r->second;
433 
434  std::list<Coord> pol;
435 
436  Coord a = connection->traci2omnet(first);
437  Coord b = connection->traci2omnet(TraCICoord(first.x, second.y));
438  Coord c = connection->traci2omnet(second);
439  Coord d = connection->traci2omnet(TraCICoord(second.x, first.y));
440 
441  pol.push_back(a);
442  pol.push_back(b);
443  pol.push_back(c);
444  pol.push_back(d);
445 
446  // draw polygon for region of interest
447  if (annotations) {
448  annotations->drawPolygon(pol, "black");
449  }
450 
451  // calculate region area
452  double ab = a.distance(b);
453  double ad = a.distance(d);
454  double area = ab * ad;
455  areaSum += area;
456  }
457 }

523  unEquippedHosts.insert(nodeId);
524  return;
525  }
526 
527  int32_t nodeVectorIndex = nextNodeVectorIndex++;
528 
529  cModule* parentmod = getParentModule();
530  if (!parentmod) throw cRuntimeError("Parent Module not found");
531 
532  cModuleType* nodeType = cModuleType::get(type.c_str());
533  if (!nodeType) throw cRuntimeError("Module Type \"%s\" not found", type.c_str());
534 
535 #if OMNETPP_BUILDNUM >= 1525
536  parentmod->setSubmoduleVectorSize(name.c_str(), nodeVectorIndex + 1);
537  cModule* mod = nodeType->create(name.c_str(), parentmod, nodeVectorIndex);
538 #else
539  // TODO: this trashes the vectsize member of the cModule, although nobody seems to use it
540  cModule* mod = nodeType->create(name.c_str(), parentmod, nodeVectorIndex, nodeVectorIndex);
541 #endif
542  mod->finalizeParameters();
543  if (displayString.length() > 0) {
544  mod->getDisplayString().parse(displayString.c_str());
545  }
546  mod->buildInside();
547  mod->scheduleStart(simTime() + updateInterval);
548 
549  preInitializeModule(mod, nodeId, position, road_id, speed, heading, signals);
550 
551  emit(traciModulePreInitSignal, mod);
552 
553  mod->callInitialize();
554  hosts[nodeId] = mod;
555 
556  // post-initialize TraCIMobility
557  auto mobilityModules = getSubmodulesOfType<TraCIMobility>(mod);
558  for (auto mm : mobilityModules) {
559  mm->changePosition();
560  }
561 
562  if (vehicleObstacleControl) {
563  std::vector<AntennaPosition> initialAntennaPositions;
564  for (auto& caModule : getSubmodulesOfType<ChannelAccess>(mod, true)) {
565  initialAntennaPositions.push_back(caModule->getAntennaPosition());
566  }
