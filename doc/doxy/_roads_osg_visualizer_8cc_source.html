45 void RoadsOsgVisualizer::initialize(int stage)
46 {
47  bool enabled = par("enabled");
48  if (!enabled) return;
49 
50  if (!hasGUI()) return;
51 
52  if (stage == 0) {
53 
54  TraCIScenarioManager* manager = TraCIScenarioManagerAccess().get();
55  ASSERT(manager);
56 
57  figures = new osg::Group();
58 
59  cOsgCanvas* canvas = getParentModule()->getOsgCanvas();
60  ASSERT(canvas);
61  osg::Group* scene = dynamic_cast<osg::Group*>(canvas->getScene());
62  if (!scene) {
63  scene = new osg::Group();
64  canvas->setScene(scene);
65  }
66  ASSERT(scene);
67  scene->addChild(figures);
68 
69  auto onTraciInitialized = [this, manager](veins::SignalPayload<bool> payload) {
70  TraCICommandInterface* traci = manager->getCommandInterface();
71  ASSERT(traci);
72 
73  std::string colorStr = par("lineColor");
74  auto color = cFigure::Color(colorStr.c_str());
75  double width = par("lineWidth");
76 
77  auto laneIds = traci->getLaneIds();
78  for (auto laneId : laneIds) {
79  auto coords = traci->lane(laneId).getShape();
80 
81  auto line = createLine(coords, color, width);
82  figures->addChild(line);
83  }
84  };
85  signalManager.subscribeCallback(manager, TraCIScenarioManager::traciInitializedSignal, onTraciInitialized);
86  }
87 }

98 osg::Geode* RoadsOsgVisualizer::createLine(const std::list<veins::Coord>& coords, cFigure::Color color, double width)
99 {
100  auto verts = new osg::Vec3Array();
101  for (auto coord : coords) {
102  verts->push_back(osg::Vec3(coord.x, coord.y, coord.z));
103  }
104 
105  auto primitiveSet = new osg::DrawArrays(osg::PrimitiveSet::LINE_STRIP);
106  primitiveSet->setFirst(0);
107  primitiveSet->setCount(verts->size());
108 
109  auto geometry = new osg::Geometry();
110  geometry->setVertexArray(verts);
111  geometry->addPrimitiveSet(primitiveSet);
112 
113  osg::Vec4 colorVec(color.red / 255.0, color.green / 255.0, color.blue / 255.0, 1.0);
114 
115  auto material = new osg::Material();
116  material->setAmbient(osg::Material::FRONT_AND_BACK, colorVec);
117  material->setDiffuse(osg::Material::FRONT_AND_BACK, colorVec);
118 
119  auto lineWidth = new osg::LineWidth();
120  lineWidth->setWidth(width);
121 
122  auto stateSet = new osg::StateSet();
123  stateSet->setAttribute(material);
124  stateSet->setMode(GL_LIGHTING, osg::StateAttribute::OFF);
125  stateSet->setAttribute(lineWidth);
126 
127  auto geode = new osg::Geode();
128  geode->addDrawable(geometry);
129  geode->setStateSet(stateSet);
130 
131  return geode;
132 }
