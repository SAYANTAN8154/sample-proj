54 void BaseWorldUtility::initializeIfNecessary()
55 {
56  if (isInitialized) return;
57 
58  use2DFlag = par("use2D");
59  playgroundSize = Coord(par("playgroundSizeX").doubleValue(), par("playgroundSizeY").doubleValue(), use2DFlag ? 0. : par("playgroundSizeZ").doubleValue());
60 
61  if (playgroundSize.x < 0) {
62  throw cRuntimeError("Playground size in X direction is invalid: (%f). Should be greater than or equal to zero.", playgroundSize.x);
63  }
64  if (playgroundSize.y < 0) {
65  throw cRuntimeError("Playground size in Y direction is invalid: (%f). Should be greater than or equal to zero.", playgroundSize.y);
66  }
67  if (!use2DFlag && playgroundSize.z < 0) {
68  throw cRuntimeError("Playground size in Z direction is invalid: (%f). Should be greater than or equal to zero.", playgroundSize.z);
69  }
70 
71  useTorusFlag = par("useTorus");
72 
73  airFrameId = 0;
74 
75  isInitialized = true;
76 }
