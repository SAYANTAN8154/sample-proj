105 void TimerSpecification::finalize()
106 {
107  switch (start_mode_) {
108  case StartMode::relative:
109  start_ += simTime();
110  start_mode_ = StartMode::absolute;
111  break;
112  case StartMode::absolute:
113  break;
114  case StartMode::immediate:
115  start_ = simTime() + period_;
116  break;
117  }
118 
119  switch (end_mode_) {
120  case EndMode::relative:
121  end_time_ += simTime();
122  end_mode_ = EndMode::absolute;
123  break;
124  case EndMode::absolute:
125  break;
126  case EndMode::repetition:
127  end_time_ = start_ + ((end_count_ - 1) * period_);
128  end_mode_ = EndMode::absolute;
129  break;
130  case EndMode::open:
131  break;
132  }
133 }

135 bool TimerSpecification::validOccurence(simtime_t time) const
136 {
137  const bool afterStart = time >= start_;
138  const bool beforeEnd = time <= end_time_;
139  const bool atPeriod = omnetpp::fmod(time - start_, period_) == 0;
140  return afterStart && (beforeEnd || end_mode_ == EndMode::open) && atPeriod;
141 }

156 bool TimerManager::handleMessage(omnetpp::cMessage* message)
157 {
158  auto* timerMessage = dynamic_cast<TimerMessage*>(message);
159  if (!timerMessage) {
160  return false;
161  }
162  ASSERT(timerMessage->isSelfMessage());
163  std::string s = timerMessage->getName();
164 
165  auto timer = timers_.find(timerMessage);
166  if (timer == timers_.end()) {
167  return false;
168  }
169  ASSERT(timer->second.valid() && timer->second.validOccurence(simTime()));
170 
171  timer->second.callback_();
172 
173  if (timers_.find(timerMessage) != timers_.end()) { // confirm that the timer has not been cancelled during the callback
174  const auto next_event = simTime() + timer->second.period_;
175  if (timer->second.validOccurence(next_event)) {
176  parent_->scheduleAt(next_event, timer->first);
177  }
178  else {
179  parent_->cancelAndDelete(timer->first);
180  timers_.erase(timer);
181  }
182  }
183 
184  return true;
185 }

187 TimerManager::TimerHandle TimerManager::create(TimerSpecification timerSpecification, const std::string name)
188 {
189  ASSERT(timerSpecification.valid());
190  timerSpecification.finalize();
191 
192  const auto ret = timers_.insert(std::make_pair(new TimerMessage(name), std::move(timerSpecification)));
193  ASSERT(ret.second);
194  parent_->scheduleAt(ret.first->second.start_, ret.first->first);
195 
196  return ret.first->first->getId();

199 void TimerManager::cancel(TimerManager::TimerHandle handle)
200 {
201  const auto entryMatchesHandle = [handle](const std::pair<TimerMessage*, TimerSpecification>& entry) { return entry.first->getId() == handle; };
202  auto timer = std::find_if(timers_.begin(), timers_.end(), entryMatchesHandle);
203  if (timer != timers_.end()) {
204  parent_->cancelAndDelete(timer->first);
205  timers_.erase(timer);
206  }
207 }
