31 void RoadsCanvasVisualizer::initialize(int stage)
32 {
33  bool enabled = par("enabled");
34  if (!enabled) return;
35 
36  if (!hasGUI()) return;
37 
38  if (stage == 0) {
39 
40  TraCIScenarioManager* manager = TraCIScenarioManagerAccess().get();
41  ASSERT(manager);
42 
43  figures = new cGroupFigure("roads");
44  double zindex = par("lineZIndex");
45 #if OMNETPP_VERSION >= 0x501
46  figures->setZIndex(zindex);
47 #endif
48  cCanvas* canvas = getParentModule()->getCanvas();
49  canvas->addFigure(figures);
50 
51  auto onTraciInitialized = [this, manager](veins::SignalPayload<bool> payload) {
52  TraCICommandInterface* traci = manager->getCommandInterface();
53  ASSERT(traci);
54 
55  std::string colorStr = par("lineColor");
56  auto color = cFigure::Color(colorStr.c_str());
57  double width = par("lineWidth");
58  bool zoom = par("lineWidthZoom");
59 
60  auto laneIds = traci->getLaneIds();
61  for (auto laneId : laneIds) {
62  auto coords = traci->lane(laneId).getShape();
63 
64  auto line = createLine(coords, color, width, zoom);
65  figures->addFigure(line);
66  }
67  };
68  signalManager.subscribeCallback(manager, TraCIScenarioManager::traciInitializedSignal, onTraciInitialized);
69  }
70 }
