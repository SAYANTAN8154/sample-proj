68 bool Obstacle::containsPoint(Coord point) const
69 {
70  bool isInside = false;
71  const Obstacle::Coords& shape = getShape();
72  Obstacle::Coords::const_iterator i = shape.begin();
73  Obstacle::Coords::const_iterator j = (shape.rbegin() + 1).base();
74  for (; i != shape.end(); j = i++) {
75  bool inYRangeUp = (point.y >= i->y) && (point.y < j->y);
76  bool inYRangeDown = (point.y >= j->y) && (point.y < i->y);
77  bool inYRange = inYRangeUp || inYRangeDown;
78  if (!inYRange) continue;
79  bool intersects = point.x < (i->x + ((point.y - i->y) * (j->x - i->x) / (j->y - i->y)));
80  if (!intersects) continue;
81  isInside = !isInside;
82  }
83  return isInside;
84 }

88 double segmentsIntersectAt(const Coord& p1From, const Coord& p1To, const Coord& p2From, const Coord& p2To)
89 {
90  double p1x = p1To.x - p1From.x;
91  double p1y = p1To.y - p1From.y;
92  double p2x = p2To.x - p2From.x;
93  double p2y = p2To.y - p2From.y;
94  double p1p2x = p1From.x - p2From.x;
95  double p1p2y = p1From.y - p2From.y;
96  double D = (p1x * p2y - p1y * p2x);
97 
98  double p1Frac = (p2x * p1p2y - p2y * p1p2x) / D;
99  if (p1Frac < 0 || p1Frac > 1) return -1;
100 
101  double p2Frac = (p1x * p1p2y - p1y * p1p2x) / D;
102  if (p2Frac < 0 || p2Frac > 1) return -1;
103 
104  return p1Frac;
105 }
106 } // namespace

108 std::vector<double> Obstacle::getIntersections(const Coord& senderPos, const Coord& receiverPos) const
109 {
110  std::vector<double> intersectAt;
111  const Obstacle::Coords& shape = getShape();
112  Obstacle::Coords::const_iterator i = shape.begin();
113  Obstacle::Coords::const_iterator j = (shape.rbegin() + 1).base();
114  for (; i != shape.end(); j = i++) {
115  const Coord& c1 = *i;
116  const Coord& c2 = *j;
117 
118  double i = segmentsIntersectAt(senderPos, receiverPos, c1, c2);
119  if (i != -1) {
120  intersectAt.push_back(i);
121  }
122  }
123  std::sort(intersectAt.begin(), intersectAt.end());
124  return intersectAt;
125 }
