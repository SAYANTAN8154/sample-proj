100 /**
101  * This function returns the logging name of the module with the
102  * specified id. It can be used for logging messages to simplify
103  * debugging in TKEnv.
104  *
105  * Only supports ids from simple module derived from the BaseModule
106  * or the nic compound module id.
107  *
108  * @return logging name of module id or NULL if not found
109  * @sa logName
110  */
111 std::string BaseModule::logName(void) const
112 {
113  std::ostringstream ost;
114  if (hasPar("logName")) { // let modules override
115  ost << par("logName").stringValue();
116  }
117  else {
118  const cModule* const parent = findHost();
119  parent->hasPar("logName") ? ost << parent->par("logName").stringValue() : ost << parent->getName();
120  ost << "[" << parent->getIndex() << "]";
121  }
122  return ost.str();
123 }
