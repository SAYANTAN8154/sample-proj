62 AnnotationManager::~AnnotationManager()
63 {
64  while (scheduledEraseEvts.begin() != scheduledEraseEvts.end()) {
65  cancelAndDelete(*scheduledEraseEvts.begin());
66  scheduledEraseEvts.erase(scheduledEraseEvts.begin());
67  }
68  scheduledEraseEvts.clear();
69 
70  while (annotations.begin() != annotations.end()) {
71  delete *annotations.begin();
72  annotations.erase(annotations.begin());
73  }
74  annotations.clear();
75 
76  while (groups.begin() != groups.end()) {
77  delete *groups.begin();
78  groups.erase(groups.begin());
79  }
80  groups.clear();
81 }

94

96  Annotation* a = static_cast<Annotation*>(msg->getContextPointer());

99  erase(a);

121 /**
122  * adds Annotations from an XML document; example below.
123  *
124  * ```
125  * <annotations>
126  * <line color="#F00" shape="16,0 8,13.8564" />
127  * <poly color="#0F0" shape="16,64 8,77.8564 -8,77.8564 -16,64 -8,50.1436 8,50.1436" />
128  * </annotations>
129  * ```
130  */
131 void AnnotationManager::addFromXml(cXMLElement* xml)
132 {
133  std::string rootTag = xml->getTagName();
134  ASSERT(rootTag == "annotations");
135 
136  cXMLElementList list = xml->getChildren();
137  for (cXMLElementList::const_iterator i = list.begin(); i != list.end(); ++i) {
138  cXMLElement* e = *i;
139 
140  std::string tag = e->getTagName();
141 
142  if (tag == "point") {
143  ASSERT(e->getAttribute("text"));
144  std::string text = e->getAttribute("text");
145  ASSERT(e->getAttribute("color"));
146  std::string color = e->getAttribute("color");
147  ASSERT(e->getAttribute("shape"));
148  std::string shape = e->getAttribute("shape");
149  std::vector<std::string> points = cStringTokenizer(shape.c_str(), " ").asVector();
150  ASSERT(points.size() == 2);
151 
152  std::vector<double> p1a = cStringTokenizer(points[0].c_str(), ",").asDoubleVector();
153  ASSERT(p1a.size() == 2);
154 
155  drawPoint(Coord(p1a[0], p1a[1]), color, text);
156  }
157  else if (tag == "line") {
158  ASSERT(e->getAttribute("color"));
159  std::string color = e->getAttribute("color");
160  ASSERT(e->getAttribute("shape"));
161  std::string shape = e->getAttribute("shape");
162  std::vector<std::string> points = cStringTokenizer(shape.c_str(), " ").asVector();
163  ASSERT(points.size() == 2);
164 
165  std::vector<double> p1a = cStringTokenizer(points[0].c_str(), ",").asDoubleVector();
166  ASSERT(p1a.size() == 2);
167  std::vector<double> p2a = cStringTokenizer(points[1].c_str(), ",").asDoubleVector();
168  ASSERT(p2a.size() == 2);
169 
170  drawLine(Coord(p1a[0], p1a[1]), Coord(p2a[0], p2a[1]), color);
171  }
172  else if (tag == "poly") {
173  ASSERT(e->getAttribute("color"));
174  std::string color = e->getAttribute("color");
175  ASSERT(e->getAttribute("shape"));
176  std::string shape = e->getAttribute("shape");
177  std::vector<std::string> points = cStringTokenizer(shape.c_str(), " ").asVector();
178  ASSERT(points.size() >= 2);
179 
180  std::vector<Coord> coords;
181  for (std::vector<std::string>::const_iterator i = points.begin(); i != points.end(); ++i) {
182  std::vector<double> pa = cStringTokenizer(i->c_str(), ",").asDoubleVector();
183  ASSERT(pa.size() == 2);
184  coords.push_back(Coord(pa[0], pa[1]));
185  }
186 
187  drawPolygon(coords, color);
188  }
189  else {
190  throw cRuntimeError("while reading annotations xml: expected 'line' or 'poly', but got '%s'", tag.c_str());
191  }
192  }
193 }

302 void AnnotationManager::show(const Annotation* annotation)
303 {
304  if (annotation->figure) return;
305 
306  if (const Point* o = dynamic_cast<const Point*>(annotation)) {
307 
308  if (hasGUI()) {
309  // no corresponding TkEnv representation
310  }
311 
312  TraCIScenarioManager* traci = TraCIScenarioManagerAccess().get();
313  if (traci && traci->isConnected()) {
314  std::stringstream nameBuilder;
315  nameBuilder << o->text << " " << getEnvir()->getUniqueNumber();
316  traci->getCommandInterface()->addPoi(nameBuilder.str(), "Annotation", TraCIColor::fromTkColor(o->color), 6, o->pos);
317  annotation->traciPoiIds.push_back(nameBuilder.str());
318  }
319  }
320  else if (const Line* l = dynamic_cast<const Line*>(annotation)) {
321 
322  if (hasGUI()) {
323  cLineFigure* figure = new cLineFigure();
324  figure->setStart(cFigure::Point(l->p1.x, l->p1.y));
325  figure->setEnd(cFigure::Point(l->p2.x, l->p2.y));
326  figure->setLineColor(cFigure::Color(l->color.c_str()));
327  annotation->figure = figure;
328  annotationLayer->addFigure(annotation->figure);
329  }
330 
331  TraCIScenarioManager* traci = TraCIScenarioManagerAccess().get();
332  if (traci && traci->isConnected()) {
333  std::list<Coord> coords;
334  coords.push_back(l->p1);
335  coords.push_back(l->p2);
336  std::stringstream nameBuilder;
337  nameBuilder << "Annotation" << getEnvir()->getUniqueNumber();
338  traci->getCommandInterface()->addPolygon(nameBuilder.str(), "Annotation", TraCIColor::fromTkColor(l->color), false, 5, coords);
339  annotation->traciLineIds.push_back(nameBuilder.str());
340  }
341  }
342  else if (const Polygon* p = dynamic_cast<const Polygon*>(annotation)) {
343 
344  ASSERT(p->coords.size() >= 2);
345 
346  if (hasGUI()) {
347  cPolygonFigure* figure = new cPolygonFigure();
348  std::vector<cFigure::Point> points;
349  for (std::list<Coord>::const_iterator i = p->coords.begin(); i != p->coords.end(); ++i) {
350  points.push_back(cFigure::Point(i->x, i->y));
351  }
352  figure->setPoints(points);
353  figure->setLineColor(cFigure::Color(p->color.c_str()));
354  figure->setFilled(false);
355  annotation->figure = figure;
356  annotationLayer->addFigure(annotation->figure);
357  }
358 
359  TraCIScenarioManager* traci = TraCIScenarioManagerAccess().get();
360  if (traci && traci->isConnected()) {
361  std::stringstream nameBuilder;
362  nameBuilder << "Annotation" << getEnvir()->getUniqueNumber();
363  traci->getCommandInterface()->addPolygon(nameBuilder.str(), "Annotation", TraCIColor::fromTkColor(p->color), false, 4, p->coords);
364  annotation->traciPolygonsIds.push_back(nameBuilder.str());
365  }
366  }
367  else {
368  throw cRuntimeError("unknown Annotation type");
369  }
370 }

372 void AnnotationManager::hide(const Annotation* annotation)
373 {
374  if (annotation->figure) {
375  delete annotationLayer->removeFigure(annotation->figure);
376  annotation->figure = nullptr;
377  }
378 
379  TraCIScenarioManager* traci = TraCIScenarioManagerAccess().get();
380  if (traci && traci->isConnected()) {
381  for (std::list<std::string>::const_iterator i = annotation->traciPolygonsIds.begin(); i != annotation->traciPolygonsIds.end(); ++i) {
382  std::string id = *i;
383  traci->getCommandInterface()->polygon(id).remove(3);
384  }
385  annotation->traciPolygonsIds.clear();
386  for (std::list<std::string>::const_iterator i = annotation->traciLineIds.begin(); i != annotation->traciLineIds.end(); ++i) {
387  std::string id = *i;
388  traci->getCommandInterface()->polygon(id).remove(4);
389  }
390  annotation->traciLineIds.clear();
391  for (std::list<std::string>::const_iterator i = annotation->traciPoiIds.begin(); i != annotation->traciPoiIds.end(); ++i) {
392  std::string id = *i;
393  traci->getCommandInterface()->poi(id).remove(5);
394  }
395  annotation->traciPoiIds.clear();
396  }
397 }
