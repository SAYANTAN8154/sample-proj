113 {
114  uint32_t length = read<uint32_t>();
115  if (length == 0) return std::string();
116  char obuf[length + 1];
117 
118  for (size_t i = 0; i < length; ++i) read<char>(obuf[i]);
119  obuf[length] = 0;
120 
121  return std::string(obuf, length);
