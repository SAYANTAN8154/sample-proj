57 double NistErrorRate::getFecBpskBer(double snr, uint32_t nbits, uint32_t bValue)
58 {
59  double ber = getBpskBer(snr);
60  if (ber == 0.0) {
61  return 1.0;
62  }
63  double pe = calculatePe(ber, bValue);

79 double NistErrorRate::calculatePe(double p, uint32_t bValue)
80 {
81  double D = std::sqrt(4.0 * p * (1.0 - p));
82  double pe = 1.0;
83  if (bValue == 1) {
84  // code rate 1/2, use table 3.1.1
85  pe = 0.5 * (36.0 * std::pow(D, 10) + 211.0 * std::pow(D, 12) + 1404.0 * std::pow(D, 14) + 11633.0 * std::pow(D, 16) + 77433.0 * std::pow(D, 18) + 502690.0 * std::pow(D, 20) + 3322763.0 * std::pow(D, 22) + 21292910.0 * std::pow(D, 24) + 134365911.0 * std::pow(D, 26));
86  }
87  else if (bValue == 2) {
88  // code rate 2/3, use table 3.1.2
89  pe = 1.0 / (2.0 * bValue) * (3.0 * std::pow(D, 6) + 70.0 * std::pow(D, 7) + 285.0 * std::pow(D, 8) + 1276.0 * std::pow(D, 9) + 6160.0 * std::pow(D, 10) + 27128.0 * std::pow(D, 11) + 117019.0 * std::pow(D, 12) + 498860.0 * std::pow(D, 13) + 2103891.0 * std::pow(D, 14) + 8784123.0 * std::pow(D, 15));
90  }
91  else if (bValue == 3) {
92  // code rate 3/4, use table 3.1.2
93  pe = 1.0 / (2.0 * bValue) * (42.0 * std::pow(D, 5) + 201.0 * std::pow(D, 6) + 1492.0 * std::pow(D, 7) + 10469.0 * std::pow(D, 8) + 62935.0 * std::pow(D, 9) + 379644.0 * std::pow(D, 10) + 2253373.0 * std::pow(D, 11) + 13073811.0 * std::pow(D, 12) + 75152755.0 * std::pow(D, 13) + 428005675.0 * std::pow(D, 14));
94  }
95  else {
96  ASSERT(false);
97  }
98  return pe;
99 }

123 double NistErrorRate::getChunkSuccessRate(unsigned int datarate, enum Bandwidth bw, double snr_mW, uint32_t nbits)
124 {
125 
126  // get mcs from datarate and bw
127  MCS mcs = getMCS(datarate, bw);
128 
129  // compute success rate depending on mcs
130  switch (mcs) {
131  case MCS::ofdm_bpsk_r_1_2:
132  return getFecBpskBer(snr_mW, nbits, 1);
133  break;
134  case MCS::ofdm_bpsk_r_3_4:
135  return getFecBpskBer(snr_mW, nbits, 3);
136  break;
137  case MCS::ofdm_qpsk_r_1_2:
138  return getFecQpskBer(snr_mW, nbits, 1);
139  break;
140  case MCS::ofdm_qpsk_r_3_4:
141  return getFecQpskBer(snr_mW, nbits, 3);
142  break;
143  case MCS::ofdm_qam16_r_1_2:
144  return getFec16QamBer(snr_mW, nbits, 1);
145  break;
146  case MCS::ofdm_qam16_r_3_4:
147  return getFec16QamBer(snr_mW, nbits, 3);
148  break;
149  case MCS::ofdm_qam64_r_2_3:
150  return getFec64QamBer(snr_mW, nbits, 2);
151  break;
152  case MCS::ofdm_qam64_r_3_4:
153  return getFec64QamBer(snr_mW, nbits, 3);
154  break;
155  default:
156  ASSERT2(false, "Invalid MCS chosen");
157  break;
158  }
159 
160  return 0;
161 }
