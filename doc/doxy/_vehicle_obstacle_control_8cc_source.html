96 Signal VehicleObstacleControl::getVehicleAttenuationSingle(double h1, double h2, double h, double d, double d1, Signal attenuationPrototype)
97 {
98  Signal attenuation = Signal(attenuationPrototype.getSpectrum());
99 
100  for (uint16_t i = 0; i < attenuation.getNumValues(); i++) {
101  double freq = attenuation.getSpectrum().freqAt(i);
102  double lambda = BaseWorldUtility::speedOfLight() / freq;
103  double d2 = d - d1;
104  double y = (h2 - h1) / d * d1 + h1;
105  double H = h - y;
106  double r1 = sqrt(lambda * d1 * d2 / d);
107  double V0 = sqrt(2) * H / r1;
108 
109  if (V0 <= -0.7) {
110  attenuation.at(i) = 0;
111  }
112  else {
113  attenuation.at(i) = 6.9 + 20 * log10(sqrt(pow((V0 - 0.1), 2) + 1) + V0 - 0.1);
114  }
115  }
116 
117  return attenuation;
118 }

120 Signal VehicleObstacleControl::getVehicleAttenuationDZ(const std::vector<std::pair<double, double>>& dz_vec, Signal attenuationPrototype)
121 {
122 
123  // basic sanity check
124  ASSERT(dz_vec.size() >= 2);
125 
126  // make sure the list of x coordinates is sorted
127  for (size_t i = 0; i < dz_vec.size() - 1; i++) {
128  ASSERT(dz_vec[i].first < dz_vec[i + 1].first);
129  }
130 
131  // find "major obstacles" (MOs) between sender and receiver via rope-stretching algorithm
132  /*
133  * |
134  * | |
135  * | : |
136  * | | : : | |
137  * mo0 mo1 mo2 mo3
138  * snd rcv
139  */
140  std::vector<size_t> mo; ///< indices of MOs (this includes the sender and receiver)
141  mo.push_back(0);
142  for (size_t i = 0;;) {
143  double max_slope = -std::numeric_limits<double>::infinity();
144  size_t max_slope_index;
145  bool have_max_slope_index = false;
146 
147  for (size_t j = i + 1; j < dz_vec.size(); ++j) {
148  double slope = (dz_vec[j].second - dz_vec[i].second) / (dz_vec[j].first - dz_vec[i].first);
149 
150  if (slope > max_slope) {
151  max_slope = slope;
152  max_slope_index = j;
153  have_max_slope_index = true;
154  }
155  }
156 
157  // Sanity check
158  ASSERT(have_max_slope_index);
159 
160  if (max_slope_index >= dz_vec.size() - 1) break;
161 
162  mo.push_back(max_slope_index);
163 
164  i = max_slope_index;
165  }
166  mo.push_back(dz_vec.size() - 1);
167 
168  // calculate attenuation due to MOs
169  Signal attenuation_mo(attenuationPrototype.getSpectrum());
170  for (size_t mm = 0; mm < mo.size() - 2; ++mm) {
171  size_t tx = mo[mm];
172  size_t ob = mo[mm + 1];
173  size_t rx = mo[mm + 2];
174 
175  double h1 = dz_vec[tx].second;
176  double h2 = dz_vec[rx].second;
177  double d = dz_vec[rx].first - dz_vec[tx].first;
178  double d1 = dz_vec[ob].first - dz_vec[tx].first;
179  double h = dz_vec[ob].second;
180 
181  Signal ad_mo = getVehicleAttenuationSingle(h1, h2, h, d, d1, attenuationPrototype);
182 
183  attenuation_mo += ad_mo;
184  }
185 
186  // calculate attenuation due to "small obstacles" (i.e. the ones in-between MOs)
187  Signal attenuation_so(attenuationPrototype.getSpectrum());
188  for (size_t i = 0; i < mo.size() - 1; ++i) {
189  size_t delta = mo[i + 1] - mo[i];
190 
191  if (delta == 1) {
192  // no obstacle in-between these two MOs
193  }
194  else if (delta == 2) {
195  // one obstacle in-between these two MOs
196  size_t tx = mo[i];
197  size_t ob = mo[i] + 1;
198  size_t rx = mo[i + 1];
199 
200  double h1 = dz_vec[tx].second;
201  double h2 = dz_vec[rx].second;
202  double d = dz_vec[rx].first - dz_vec[tx].first;
203  double d1 = dz_vec[ob].first - dz_vec[tx].first;
204  double h = dz_vec[ob].second;
205 
206  Signal ad_mo = getVehicleAttenuationSingle(h1, h2, h, d, d1, attenuationPrototype);
207  attenuation_so += ad_mo;
208  }
209  else {
210  // multiple obstacles in-between these two MOs -- use the one closest to their line of sight
211  double x1 = dz_vec[mo[i]].first;
212  double y1 = dz_vec[mo[i]].second;
213  double x2 = dz_vec[mo[i + 1]].first;
214  double y2 = dz_vec[mo[i + 1]].second;
215 
216  double min_delta_h = std::numeric_limits<float>::infinity();
217  size_t min_delta_h_index;
218  bool have_min_delta_h_index = false;
219  for (size_t j = mo[i] + 1; j < mo[i + 1]; ++j) {
220  double h = (y2 - y1) / (x2 - x1) * (dz_vec[j].first - x1) + y1;
221  double delta_h = h - dz_vec[j].second;
222 
223  if (delta_h < min_delta_h) {
224  min_delta_h = delta_h;
225  min_delta_h_index = j;
226  have_min_delta_h_index = true;
227  }
228  }
229 
230  // Sanity check
231  ASSERT(have_min_delta_h_index);
232 
233  size_t tx = mo[i];
234  size_t ob = min_delta_h_index;
235  size_t rx = mo[i + 1];
236 
237  double h1 = dz_vec[tx].second;
238  double h2 = dz_vec[rx].second;
239  double d = dz_vec[rx].first - dz_vec[tx].first;
240  double d1 = dz_vec[ob].first - dz_vec[tx].first;
241  double h = dz_vec[ob].second;
242 
243  Signal ad_mo = getVehicleAttenuationSingle(h1, h2, h, d, d1, attenuationPrototype);
244  attenuation_so += ad_mo;
245  }
246  }
247 
248  double c;
249  {
250  double prodS = 1;
251  double sumS = 0;
252  double prodSsum = 1;
253  double firstS = 0;
254  double lastS = 0;
255 
256  double s_old = 0;
257  for (size_t jj = 0; jj < mo.size() - 1; ++jj) {
258  double s = dz_vec[mo[jj + 1]].first - dz_vec[mo[jj]].first; ///< distance between two MOs
259 
260  prodS *= s;
261  sumS += s;
262  if (jj == 0)
263  firstS = s;
264  else if (jj > 0)
265  prodSsum *= (s + s_old);
266  if (jj == mo.size() - 2) lastS = s;
267  s_old = s;
268  }
269 
270  c = -10 * log10((prodS * sumS) / (prodSsum * firstS * lastS));
271  }
272 
273  return attenuation_mo + attenuation_so + c;
274 }

276 std::vector<std::pair<double, double>> VehicleObstacleControl::getPotentialObstacles(const AntennaPosition& senderPos_, const AntennaPosition& receiverPos_, const Signal& s) const
277 {
278  Enter_Method_Silent();
279 
280  auto senderPos = senderPos_.getPositionAt();
281  auto receiverPos = receiverPos_.getPositionAt();
282 
283  double senderHeight = senderPos.z;
284  double receiverHeight = receiverPos.z;
285 
286  ASSERT(senderHeight > 0);
287  ASSERT(receiverHeight > 0);
288 
289  std::vector<std::pair<double, double>> potentialObstacles; /**< linear position of each obstructing vehicle along (senderPos--receiverPos) */
290 
291  simtime_t sStart = s.getSendingStart();
292 
293  EV << "searching candidates for transmission from " << senderPos.info() << " -> " << receiverPos.info() << " (" << senderPos.distance(receiverPos) << "meters total)" << std::endl;
294 
295  if (hasGUI() && annotations) {
296  annotations->eraseAll(vehicleAnnotationGroup);
297  drawVehicleObstacles(sStart);
298  annotations->drawLine(senderPos, receiverPos, "blue", vehicleAnnotationGroup);
299  }
300 
301  double x1 = std::min(senderPos.x, receiverPos.x);
302  double x2 = std::max(senderPos.x, receiverPos.x);
303  double y1 = std::min(senderPos.y, receiverPos.y);
304  double y2 = std::max(senderPos.y, receiverPos.y);
305 
306  for (auto o : vehicleObstacles) {
307  auto obstacleAntennaPositions = o->getInitialAntennaPositions();
308  double l = o->getLength();
309  double w = o->getWidth();
310  double h = o->getHeight();
311 
312  auto hp_approx = o->getMobility()->getPositionAt(simTime());
313 
314  EV << "checking vehicle in proximity of " << hp_approx.info() << " with height: " << h << " width: " << w << " length: " << l << endl;
315 
316  if (!o->maybeInBounds(x1, y1, x2, y2, sStart)) {
317  EV_TRACE << "bounding boxes don't overlap: ignore" << std::endl;
318  continue;
319  }
320 
321  // check if this is either the sender or the receiver
322  bool ignoreMe = false;
323  for (auto obstacleAntenna : obstacleAntennaPositions) {
324  if (obstacleAntenna.isSameAntenna(senderPos_)) {
325  EV_TRACE << "...this is the sender: ignore" << std::endl;
326  ignoreMe = true;
327  }
328  if (obstacleAntenna.isSameAntenna(receiverPos_)) {
329  EV_TRACE << "...this is the receiver: ignore" << std::endl;
330  ignoreMe = true;
331  }
332  }
333  if (ignoreMe) continue;
334 
335  // this is a potential obstacle
336  double p1d = o->getIntersectionPoint(senderPos, receiverPos, sStart);
337  double maxd = senderPos.distance(receiverPos);
338  if (!std::isnan(p1d) && p1d > 0 && p1d < maxd) {
339  auto it = potentialObstacles.begin();
340  while (true) {
341  if (it == potentialObstacles.end()) {
342  potentialObstacles.emplace_back(p1d, h);
343  break;
344  }
345  if (it->first == p1d) { // omit double entries
346  EV << "two obstacles at same distance " << it->first << " == " << p1d << " height: " << it->second << " =? " << h << std::endl;
347  break;
348  }
349  if (it->first > p1d) {
350  potentialObstacles.insert(it, std::make_pair(p1d, h));
351  break;
352  }
353  ++it;
354  }
355  EV << "\tgot obstacle in 2d-LOS, " << p1d << " meters away from sender" << std::endl;
356  Coord hitPos = senderPos + (receiverPos - senderPos) / senderPos.distance(receiverPos) * p1d;
357  if (hasGUI() && annotations) {
358  annotations->drawLine(senderPos, hitPos, "red", vehicleAnnotationGroup);
359  }
360  }
361  }
362 
363  return potentialObstacles;
364 }
