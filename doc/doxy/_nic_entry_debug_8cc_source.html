47 void NicEntryDebug::disconnectFrom(NicEntry* other)
48 {
49  EV_TRACE << "disconnecting nic #" << nicId << " and #" << other->nicId << endl;
50 
51  NicEntryDebug* otherNic = (NicEntryDebug*) other;
52 
53  // search the connection in the outConns list
54  GateList::iterator p = outConns.find(other);
55  // no need to check whether entry is valid; is already check by ConnectionManager isConnected
56  // get the hostGate
57  // order is phyGate->nicGate->hostGate
58  cGate* hostGate = p->second->getNextGate()->getNextGate();
59 
60  // release local out gate
61  freeOutGates.push_back(hostGate);
62 
63  // release remote in gate
64  otherNic->freeInGates.push_back(hostGate->getNextGate());
65 
66  // reset gates
67  // hostGate->getNextGate()->connectTo(0);
68  hostGate->disconnect();
69 
70  // delete the connection
71  outConns.erase(p);
72 }

74 int NicEntryDebug::collectGates(const char* pattern, GateStack& gates)
75 {
76  cModule* host = nicPtr->getParentModule();
77  int i = 1;
78  char gateName[20];
79  // create the unique name for the gate (composed of the nic module id and a counter)
80  sprintf(gateName, pattern, nicId, i);
81  while (host->hasGate(gateName)) {
82  cGate* hostGate = host->gate(gateName);
83  if (hostGate->isConnectedOutside()) {
84  throw cRuntimeError("Gate %s is still connected but not registered with this NicEntry. Either the last NicEntry for this NIC did not clean up correctly or another gate creation module is interfering with this one!", gateName);
85  }
86  ASSERT(hostGate->isConnectedInside());
87  gates.push_back(hostGate);
88 
89  ++i;
90  sprintf(gateName, pattern, nicId, i);
91  }
92 
93  return i - 1;
94 }

109 cGate* NicEntryDebug::requestInGate(void)
110 {
111  collectFreeGates();
112 
113  // gate of the host
114  cGate* hostGate;
115 
116  if (!freeInGates.empty()) {
117  hostGate = freeInGates.back();
118  freeInGates.pop_back();
119  }
120  else {
121  char gateName[20];
122 
123  // we will have one more in gate
124  ++inCnt;
125 
126  // get a unique name for the gate (composed of the nic module id and a counter)
127  sprintf(gateName, "in%d-%d", nicId, inCnt);
128 
129  // create a new gate for the host module
130  nicPtr->getParentModule()->addGate(gateName, cGate::INPUT);
131  hostGate = nicPtr->getParentModule()->gate(gateName);
132 
133  // gate of the nic
134  cGate* nicGate;
135 
136  // create a new gate for the nic module
137  nicPtr->addGate(gateName, cGate::INPUT);
138  nicGate = nicPtr->gate(gateName);
139 
140  // connect the hist gate with the nic gate
141  hostGate->connectTo(nicGate);
142 
143  // pointer to the phy module
144  ChannelAccess* phyModule;
145  // gate of the phy module
146  cGate* phyGate;
147 
148  // to avoid unnecessary dynamic_casting we check for a "phy"-named submodule first
149  if ((phyModule = dynamic_cast<ChannelAccess*>(nicPtr->getSubmodule("phy"))) == nullptr) phyModule = FindModule<ChannelAccess*>::findSubModule(nicPtr);
150  ASSERT(phyModule != nullptr);
151 
152  // create a new gate for the phy module
153  phyModule->addGate(gateName, cGate::INPUT);
154  phyGate = phyModule->gate(gateName);
155 
156  // connect the nic gate (the gate of the compound module) to
157  // a "real" gate -- the gate of the phy module
158  nicGate->connectTo(phyGate);
159  }
160 
161  return hostGate;
162 }
