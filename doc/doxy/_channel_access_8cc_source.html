83 void ChannelAccess::sendToChannel(cPacket* msg)
84 {
85  EV_TRACE << "sendToChannel: sending to gates\n";
86 
87  const auto& gateList = cc->getGateList(getParentModule()->getId());
88 
89  for (auto&& entry : gateList) {
90  const auto gate = entry.second;
91  const auto propagationDelay = calculatePropagationDelay(entry.first);
92 
93  if (useSendDirect) {
94  if (gate->isVector()) {
95  for (int gateIndex = gate->getBaseId(); gateIndex < gate->getBaseId() + gate->size(); gateIndex++) {
96  sendDirect(msg->dup(), propagationDelay, msg->getDuration(), gate->getOwnerModule(), gateIndex);
97  }
98  }
99  else {
100  sendDirect(msg->dup(), propagationDelay, msg->getDuration(), gate->getOwnerModule(), gate->getBaseId());
101  }
102  }
103  else {
104  sendDelayed(msg->dup(), propagationDelay, gate);
105  }
106  }
107  // Original message no longer needed, copies have been sent to all possible receivers.
108  delete msg;
109 }

111 simtime_t ChannelAccess::calculatePropagationDelay(const NicEntry* nic)
112 {
113  if (!usePropagationDelay) return 0;
114 
115  ChannelAccess* const senderModule = this;
116  ChannelAccess* const receiverModule = nic->chAccess;
117  // const simtime_t_cref sStart = simTime();
118 
119  ASSERT(senderModule);
120  ASSERT(receiverModule);
121 
122  /** claim the Move pattern of the sender from the Signal */
123  Coord senderPos = senderModule->antennaPosition.getPositionAt();
124  Coord receiverPos = receiverModule->antennaPosition.getPositionAt();
125 
126  // this time-point is used to calculate the distance between sending and receiving host
127  return receiverPos.distance(senderPos) / BaseWorldUtility::speedOfLight();
128 }

130 void ChannelAccess::receiveSignal(cComponent* source, simsignal_t signalID, cObject* obj, cObject* details)
131 {
132  if (signalID == BaseMobility::mobilityStateChangedSignal) {
133  ChannelMobilityPtrType const mobility = check_and_cast<ChannelMobilityPtrType>(obj);
134 
135  auto heading = Heading::fromCoord(mobility->getCurrentOrientation());
136  antennaPosition = AntennaPosition(getId(), mobility->getPositionAt(simTime()) + antennaOffset.rotatedYaw(-heading.getRad()), mobility->getCurrentSpeed(), simTime());
137  antennaHeading = Heading(heading.getRad() + antennaOffsetYaw);
138 
139  if (isRegistered) {
140  cc->updateNicPos(getParentModule()->getId(), antennaPosition.getPositionAt(), antennaHeading);
141  }
142  else {
143  // register the nic with ConnectionManager
144  // returns true, if sendDirect is used
145  useSendDirect = cc->registerNic(getParentModule(), this, antennaPosition.getPositionAt(), antennaHeading);
146  isRegistered = true;
147  }
148  }
149 }
