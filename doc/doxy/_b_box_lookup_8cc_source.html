32 /**
33  * Helper structure representing a wireless ray from a sender to a receiver.
34  *
35  * Contains pre-computed values to speed up calls to intersect with the same ray but different boxes.
36  */
37 struct Ray {
38  Point origin;
39  Point destination;
40  Point direction;
41  Point invDirection;
42  struct {
43  size_t x;
44  size_t y;
45  } sign;
46  double length;
47 };
48 
49 /**
50  * Return a Ray struct for fast intersection tests from sender to receiver.
51  */
52 Ray makeRay(const Point& sender, const Point& receiver)
53 {
54  const double dir_x = receiver.x - sender.x;
55  const double dir_y = receiver.y - sender.y;
56  Ray ray;
57  ray.origin = sender;
58  ray.destination = receiver;
59  ray.length = std::sqrt(dir_x * dir_x + dir_y * dir_y);
60  ray.direction.x = dir_x / ray.length;
61  ray.direction.y = dir_y / ray.length;
62  ray.invDirection.x = 1.0 / ray.direction.x;
63  ray.invDirection.y = 1.0 / ray.direction.y;
64  ray.sign.x = ray.invDirection.x < 0;
65  ray.sign.y = ray.invDirection.y < 0;
66  return ray;
67 }
68 /**
69  * Return whether ray intersects with box.
70  *
71  * Based on:
72  * Amy Williams, Steve Barrus, R. Keith Morley & Peter Shirley (2005) An Efficient and Robust Ray-Box Intersection Algorithm, Journal of Graphics Tools, 10:1, 49-54, DOI: 10.1080/2151237X.2005.10129188
73  */
74 bool intersects(const Ray& ray, const Box& box)
75 {
76  const double x[2]{box.p1.x, box.p2.x};
77  const double y[2]{box.p1.y, box.p2.y};
78  double tmin = (x[ray.sign.x] - ray.origin.x) * ray.invDirection.x;
79  double tmax = (x[1 - ray.sign.x] - ray.origin.x) * ray.invDirection.x;
80  double tymin = (y[ray.sign.y] - ray.origin.y) * ray.invDirection.y;
81  double tymax = (y[1 - ray.sign.y] - ray.origin.y) * ray.invDirection.y;
82 
83  if ((tmin > tymax) || (tymin > tmax)) return false;
84  if (tymin > tmin) tmin = tymin;
85  if (tymax < tmax) tmax = tymax;
86  return (tmin < ray.length) && (tmax > 0);
87 }

93 BBoxLookup::BBoxLookup(const std::vector<Obstacle*>& obstacles, std::function<BBoxLookup::Box(Obstacle*)> makeBBox, double scenarioX, double scenarioY, int cellSize)
94  : bboxes()
95  , obstacleLookup()
96  , bboxCells()
97  , cellSize(cellSize)
98  , numCols(std::floor(scenarioX / cellSize) + 1)
99  , numRows(std::floor(scenarioY / cellSize) + 1)
100 {
101  // phase 1: build unordered collection of cells
102  // initialize proto-cells (cells in non-contiguos memory)
103  ASSERT(scenarioX > 0);
104  ASSERT(scenarioY > 0);
105  ASSERT(numCols * cellSize >= scenarioX);
106  ASSERT(numRows * cellSize >= scenarioY);
107  const size_t numCells = numCols * numRows;
108  std::vector<std::vector<BBoxLookup::Box>> protoCells(numCells);
109  std::vector<std::vector<Obstacle*>> protoLookup(numCells);
110  // fill protoCells with boundingBoxes
111  size_t numEntries = 0;
112  for (const auto obstaclePtr : obstacles) {
113  auto bbox = makeBBox(obstaclePtr);
114  const size_t fromCol = std::min(size_t(std::max(0, int(bbox.p1.x / cellSize))), numCols - 1);
115  const size_t toCol = std::min(size_t(std::max(0, int(bbox.p2.x / cellSize))), numCols - 1);
116  const size_t fromRow = std::min(size_t(std::max(0, int(bbox.p1.y / cellSize))), numRows - 1);
117  const size_t toRow = std::min(size_t(std::max(0, int(bbox.p2.y / cellSize))), numRows - 1);
118  for (size_t row = fromRow; row <= toRow; ++row) {
119  for (size_t col = fromCol; col <= toCol; ++col) {
120  ASSERT(row >= 0);
121  ASSERT(col >= 0);
122  ASSERT(row < numRows);
123  ASSERT(col < numCols);
124  const size_t cellIndex = col + row * numCols;
125  protoCells[cellIndex].push_back(bbox);
126  protoLookup[cellIndex].push_back(obstaclePtr);
127  ++numEntries;
128  ASSERT(protoCells[cellIndex].size() == protoLookup[cellIndex].size());
129  }
130  }
131  }
132 
133  // phase 2: derive read-only data structure with fast lookup
134  bboxes.reserve(numEntries);
135  obstacleLookup.reserve(numEntries);
136  bboxCells.reserve(numCells);
137  size_t index = 0;
138  for (size_t row = 0; row < numRows; ++row) {
139  for (size_t col = 0; col < numCols; ++col) {
140  const size_t cellIndex = col + row * numCols;
141  auto& currentCell = protoCells.at(cellIndex);
142  auto& currentLookup = protoLookup.at(cellIndex);
143  ASSERT(currentCell.size() == currentLookup.size());
144  const size_t count = currentCell.size();
145  // copy over bboxes and obstacle lookups (in strict order)
146  for (size_t entryIndex = 0; entryIndex < count; ++entryIndex) {
147  bboxes.push_back(currentCell.at(entryIndex));
148  obstacleLookup.push_back(currentLookup.at(entryIndex));
149  }
150  // create lookup table for this cell
151  bboxCells.push_back({index, count});
152  // forward index to begin of next cell
153  index += count;
154  ASSERT(bboxes.size() == index);
155  }
156  }
157  ASSERT(bboxes.size() == numEntries);
158  ASSERT(bboxes.size() == obstacleLookup.size());
159 }

161 std::vector<Obstacle*> BBoxLookup::findOverlapping(Point sender, Point receiver) const
162 {
163  std::vector<Obstacle*> overlappingObstacles;
164  const Box bbox{
165  {std::min(sender.x, receiver.x), std::min(sender.y, receiver.y)},
166  {std::max(sender.x, receiver.x), std::max(sender.y, receiver.y)},
167  };
168 
169  // determine coordinates for all cells touched by bbox
170  const size_t firstCol = std::min(size_t(std::max(0, int(bbox.p1.x / cellSize))), numCols - 1);
171  const size_t lastCol = std::min(size_t(std::max(0, int(bbox.p2.x / cellSize))), numCols - 1);
172  const size_t firstRow = std::min(size_t(std::max(0, int(bbox.p1.y / cellSize))), numRows - 1);
173  const size_t lastRow = std::min(size_t(std::max(0, int(bbox.p2.y / cellSize))), numRows - 1);
174  ASSERT(lastCol < numCols && lastRow < numRows);
175  // precompute transmission ray properties
176  const Ray ray = makeRay(sender, receiver);
177  // iterate over cells
178  for (size_t row = firstRow; row <= lastRow; ++row) {
179  for (size_t col = firstCol; col <= lastCol; ++col) {
180  // skip cell if ray does not intersect with the cell.
181  const Box cellBox = {{static_cast<double>(col * cellSize), static_cast<double>(row * cellSize)}, {static_cast<double>((col + 1) * cellSize), static_cast<double>((row + 1) * cellSize)}};
182  if (!intersects(ray, cellBox)) continue;
183  // derive cell for current cell coordinates
184  const size_t cellIndex = col + row * numCols;
185  const BBoxCell& cell = bboxCells.at(cellIndex);
186  // iterate over bboxes in each cell
187  for (size_t bboxIndex = cell.index; bboxIndex < cell.index + cell.count; ++bboxIndex) {
188  const Box& current = bboxes.at(bboxIndex);
189  // check for overlap with bbox (fast rejection)
190  if (current.p2.x < bbox.p1.x) continue;
191  if (current.p1.x > bbox.p2.x) continue;
192  if (current.p2.y < bbox.p1.y) continue;
193  if (current.p1.y > bbox.p2.y) continue;
194  // derive corresponding obstacle
195  if (!intersects(ray, current)) continue;
196  overlappingObstacles.push_back(obstacleLookup.at(bboxIndex));
197  }
198  }
199  }
200  return overlappingObstacles;
201 }
