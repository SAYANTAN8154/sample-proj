76 void BaseMobility::initialize(int stage)
77 {
78  BaseModule::initialize(stage);
79  if (stage == 0) {
80  EV_TRACE << "initializing BaseMobility stage " << stage << endl;
81 
82  hasPar("scaleNodeByDepth") ? scaleNodeByDepth = par("scaleNodeByDepth").boolValue() : scaleNodeByDepth = true;
83 
84  // get utility pointers (world and host)
85  world = FindModule<BaseWorldUtility*>::findGlobalModule();
86  if (world == nullptr) throw cRuntimeError("Could not find BaseWorldUtility module");
87 
88  EV_TRACE << "initializing BaseUtility stage " << stage << endl; // for node position
89 
90  if (hasPar("updateInterval")) {
91  updateInterval = par("updateInterval");
92  }
93  else {
94  updateInterval = 0;
95  }
96 
97  // initialize Move parameter
98  bool use2D = world->use2D();
99 
100  // initalize position with random values
101  Coord pos = world->getRandomPosition();
102 
103  if (hasStartPosition) {
104  pos = startPosition;
105  }
106  else {
107  // read coordinates from parameters if available
108  double x = hasPar("x") ? par("x").doubleValue() : -1;
109  double y = hasPar("y") ? par("y").doubleValue() : -1;
110  double z = hasPar("z") ? par("z").doubleValue() : -1;
111 
112  // set position with values from parameters if available
113  if (x > -1) pos.x = x;
114  if (y > -1) pos.y = y;
115  if (!use2D && z > -1) pos.z = z;
116  }
117 
118  if (!hasPar("xOrientation") || !hasPar("yOrientation")) {
119  throw cRuntimeError("Orientation coordinates in x and y direction have to specified (necessary for antenna gain calculation)");
120  }
121  else {
122  double zOrient = hasPar("zOrientation") ? par("zOrientation").doubleValue() : 0;
123  Coord orient(par("xOrientation").doubleValue(), par("yOrientation").doubleValue(), zOrient);
124  move.setOrientationByVector(orient);
125  }
126 
127  // set start-position and start-time (i.e. current simulation-time) of the Move
128  move.setStart(pos);
129  EV_TRACE << "start pos: " << move.getStartPos().info() << endl;
130 
131  // check whether position is within the playground
132  if (!isInBoundary(move.getStartPos(), Coord::ZERO, *world->getPgs())) {
133  throw cRuntimeError("node position specified in omnetpp.ini exceeds playgroundsize");
134  }
135 
136  // set speed and direction of the Move
137  move.setSpeed(0);
138  move.setDirectionByVector(Coord::ZERO);
139  }
140  else if (stage == 1) {
141  EV_TRACE << "initializing BaseMobility stage " << stage << endl;
142 
143  // get playground scaling
144  if (world->getParentModule() != nullptr) {
145  const cDisplayString& dispWorldOwner = world->getParentModule()->getDisplayString();
146 
147  if (dispWorldOwner.containsTag("bgb")) {
148  double origPGWidth = 0.0;
149  double origPGHeight = 0.0;
150  // normally this should be equal to playground size
151  std::istringstream(dispWorldOwner.getTagArg("bgb", 0)) >> origPGWidth;
152  std::istringstream(dispWorldOwner.getTagArg("bgb", 1)) >> origPGHeight;
153 
154  // bgb of zero means size isn't set manually
155  if (origPGWidth > 0) {
156  playgroundScaleX = origPGWidth / playgroundSizeX();
157  }
158  if (origPGHeight > 0) {
159  playgroundScaleY = origPGHeight / playgroundSizeY();
160  }
161  }
162  }
163 
164  // get original display of host
165  cDisplayString& disp = findHost()->getDisplayString();
166 
167  // get host width and height
168  if (disp.containsTag("b")) {
169  std::istringstream(disp.getTagArg("b", 0)) >> origDisplayWidth;
170  std::istringstream(disp.getTagArg("b", 1)) >> origDisplayHeight;
171  }
172  // get hosts icon size
173  if (disp.containsTag("i")) {
174  // choose a appropriate icon size (only if a icon is specified)
175  origIconSize = iconSizeTagToSize(disp.getTagArg("is", 0));
176  }
177 
178  // print new host position on the screen and update bb info
179  updatePosition();
180 
181  if (move.getSpeed() > 0 && updateInterval > 0) {
182  EV_TRACE << "Host is moving, speed=" << move.getSpeed() << " updateInterval=" << updateInterval << endl;
183  moveMsg = new cMessage("move", MOVE_HOST);
184  // host moves the first time after some random delay to avoid synchronized movements
185  scheduleAt(simTime() + uniform(0, updateInterval), moveMsg);
186  }
187  }
188 }

258 void BaseMobility::handleBorderMsg(cMessage* msg)
259 {
260  EV_TRACE << "start MOVE_TO_BORDER:" << move.info() << endl;
261 
262  BorderMsg* bMsg = static_cast<BorderMsg*>(msg);
263 
264  switch (bMsg->getPolicy()) {
265  case REFLECT:
266  move.setStart(bMsg->getStartPos());
267  move.setDirectionByVector(bMsg->getDirection());
268  break;
269 
270  case WRAP:
271  move.setStart(bMsg->getStartPos());
272  break;
273 
274  case PLACERANDOMLY:
275  move.setStart(bMsg->getStartPos());
276  EV_TRACE << "new random position: " << move.getStartPos().info() << endl;
277  break;
278 
279  case RAISEERROR:
280  throw cRuntimeError("node moved outside the playground");
281  break;
282 
283  default:
284  throw cRuntimeError("Unknown BorderPolicy!");
285  break;
286  }
287 
288  fixIfHostGetsOutside();
289 
290  updatePosition();
291 
292  delete bMsg;
293 
294  EV_TRACE << "end MOVE_TO_BORDER:" << move.info() << endl;
295 }

297 void BaseMobility::updatePosition()
298 {
299  EV_DEBUG << "updatePosition: " << move.info() << endl;
300 
301  // publish the the new move
302  emit(mobilityStateChangedSignal, this);
303 
304  if (hasGUI()) {
305  std::ostringstream osDisplayTag;
306 #ifdef __APPLE__
307  const int iPrecis = 0;
308 #else
309  const int iPrecis = 5;
310 #endif
311  cDisplayString& disp = findHost()->getDisplayString();
312 
313  // setup output stream
314  osDisplayTag << std::fixed;
315  osDisplayTag.precision(iPrecis);
316 
317  if (playgroundScaleX != 1.0) {
318  osDisplayTag << (move.getStartPos().x * playgroundScaleX);
319  }
320  else {
321  osDisplayTag << (move.getStartPos().x);
322  }
323  disp.setTagArg("p", 0, osDisplayTag.str().data());
324 
325  osDisplayTag.str(""); // reset
326  if (playgroundScaleY != 1.0) {
327  osDisplayTag << (move.getStartPos().y * playgroundScaleY);
328  }
329  else {
330  osDisplayTag << (move.getStartPos().y);
331  }
332  disp.setTagArg("p", 1, osDisplayTag.str().data());
333 
334  if (!world->use2D() && scaleNodeByDepth) {
335  const double minScale = 0.25;
336  const double maxScale = 1.0;
337 
338  // scale host dependent on their z coordinate to simulate a depth
339  // effect
340  // z-coordinate of zero maps to a scale of maxScale (very close)
341  // z-coordinate of playground size z maps to size of minScale (far away)
342  double depthScale = minScale + (maxScale - minScale) * (1.0 - move.getStartPos().z / playgroundSizeZ());
343 
344  if (origDisplayWidth > 0.0 && origDisplayHeight > 0.0) {
345  osDisplayTag.str(""); // reset
346  osDisplayTag << (origDisplayWidth * depthScale);
347  disp.setTagArg("b", 0, osDisplayTag.str().data());
348 
349  osDisplayTag.str(""); // reset
350  osDisplayTag << (origDisplayHeight * depthScale);
351  disp.setTagArg("b", 1, osDisplayTag.str().data());
352  }
353 
354  if (origIconSize > 0) {
355  // choose a appropriate icon size (only if a icon is specified)
356  disp.setTagArg("is", 0, iconSizeToTag(origIconSize * depthScale));
357  }
358  }
359  }

362 void BaseMobility::reflectCoordinate(BorderHandling border, Coord& c)
363 {
364  switch (border) {
365  case X_SMALLER:
366  c.x = (-c.x);
367  break;
368  case X_BIGGER:
369  c.x = (2 * playgroundSizeX() - c.x);
370  break;
371 
372  case Y_SMALLER:
373  c.y = (-c.y);
374  break;
375  case Y_BIGGER:
376  c.y = (2 * playgroundSizeY() - c.y);
377  break;
378 
379  case Z_SMALLER:
380  c.z = (-c.z);
381  break;
382  case Z_BIGGER:
383  c.z = (2 * playgroundSizeZ() - c.z);
384  break;
385 
386  case NOWHERE:
387  default:
388  throw cRuntimeError("wrong border handling case!");
389  break;
390  }
391 }

458 BaseMobility::BorderHandling BaseMobility::checkIfOutside(Coord targetPos, Coord& borderStep)
459 {
460  BorderHandling outside = NOWHERE;
461 
462  // Testing x-value
463  if (targetPos.x < 0) {
464  borderStep.x = (-move.getStartPos().x);
465  outside = X_SMALLER;
466  }
467  else if (targetPos.x >= playgroundSizeX()) {
468  borderStep.x = (playgroundSizeX() - move.getStartPos().x);
469  outside = X_BIGGER;
470  }
471 
472  // Testing y-value
473  if (targetPos.y < 0) {
474  borderStep.y = (-move.getStartPos().y);
475 
476  if (outside == NOWHERE || fabs(borderStep.x / move.getDirection().x) > fabs(borderStep.y / move.getDirection().y)) {
477  outside = Y_SMALLER;
478  }
479  }
480  else if (targetPos.y >= playgroundSizeY()) {
481  borderStep.y = (playgroundSizeY() - move.getStartPos().y);
482 
483  if (outside == NOWHERE || fabs(borderStep.x / move.getDirection().x) > fabs(borderStep.y / move.getDirection().y)) {
484  outside = Y_BIGGER;
485  }
486  }
487 
488  // Testing z-value
489  if (!world->use2D()) {
490  // going to reach the lower z-border
491  if (targetPos.z < 0) {
492  borderStep.z = (-move.getStartPos().z);
493 
494  // no border reached so far
495  if (outside == NOWHERE) {
496  outside = Z_SMALLER;
497  }
498  // an y-border is reached earliest so far, test whether z-border
499  // is reached even earlier
500  else if ((outside == Y_SMALLER || outside == Y_BIGGER) && fabs(borderStep.y / move.getDirection().y) > fabs(borderStep.z / move.getDirection().z)) {
501  outside = Z_SMALLER;
502  }
503  // an x-border is reached earliest so far, test whether z-border
504  // is reached even earlier
505  else if ((outside == X_SMALLER || outside == X_BIGGER) && fabs(borderStep.x / move.getDirection().x) > fabs(borderStep.z / move.getDirection().z)) {
506  outside = Z_SMALLER;
507  }
508  }
509  // going to reach the upper z-border
510  else if (targetPos.z >= playgroundSizeZ()) {
511  borderStep.z = (playgroundSizeZ() - move.getStartPos().z);
512 
513  // no border reached so far
514  if (outside == NOWHERE) {
515  outside = Z_BIGGER;
516  }
517  // an y-border is reached earliest so far, test whether z-border
518  // is reached even earlier
519  else if ((outside == Y_SMALLER || outside == Y_BIGGER) && fabs(borderStep.y / move.getDirection().y) > fabs(borderStep.z / move.getDirection().z)) {
520  outside = Z_BIGGER;
521  }
522  // an x-border is reached earliest so far, test whether z-border
523  // is reached even earlier
524  else if ((outside == X_SMALLER || outside == X_BIGGER) && fabs(borderStep.x / move.getDirection().x) > fabs(borderStep.z / move.getDirection().z)) {
525  outside = Z_BIGGER;
526  }
527  }
528  }
529 
530  EV_TRACE << "checkIfOutside, outside=" << outside << " borderStep: " << borderStep.info() << endl;
531 
532  return outside;
533 }

535 void BaseMobility::goToBorder(BorderPolicy policy, BorderHandling wo, Coord& borderStep, Coord& borderStart)
536 {
537  double factor;
538 
539  EV_TRACE << "goToBorder: startPos: " << move.getStartPos().info() << " borderStep: " << borderStep.info() << " BorderPolicy: " << policy << " BorderHandling: " << wo << endl;
540 
541  switch (wo) {
542  case X_SMALLER:
543  factor = borderStep.x / move.getDirection().x;
544  borderStep.y = (factor * move.getDirection().y);
545  if (!world->use2D()) {
546  borderStep.z = (factor * move.getDirection().z); // 3D case
547  }
548 
549  if (policy == WRAP) {
550  borderStart.x = (playgroundSizeX());
551  borderStart.y = (move.getStartPos().y + borderStep.y);
552  if (!world->use2D()) {
553  borderStart.z = (move.getStartPos().z + borderStep.z); // 3D case
554  }
555  }
556  break;
557 
558  case X_BIGGER:
559  factor = borderStep.x / move.getDirection().x;
560  borderStep.y = (factor * move.getDirection().y);
561  if (!world->use2D()) {
562  borderStep.z = (factor * move.getDirection().z); // 3D case
563  }
564 
565  if (policy == WRAP) {
566  borderStart.x = (0);
567  borderStart.y = (move.getStartPos().y + borderStep.y);
568  if (!world->use2D()) {
569  borderStart.z = (move.getStartPos().z + borderStep.z); // 3D case
570  }
571  }
572  break;
573 
574  case Y_SMALLER:
575  factor = borderStep.y / move.getDirection().y;
576  borderStep.x = (factor * move.getDirection().x);
577  if (!world->use2D()) {
578  borderStep.z = (factor * move.getDirection().z); // 3D case
579  }
580 
581  if (policy == WRAP) {
582  borderStart.y = (playgroundSizeY());
583  borderStart.x = (move.getStartPos().x + borderStep.x);
584  if (!world->use2D()) {
585  borderStart.z = (move.getStartPos().z + borderStep.z); // 3D case
586  }
587  }
588  break;
589 
590  case Y_BIGGER:
591  factor = borderStep.y / move.getDirection().y;
592  borderStep.x = (factor * move.getDirection().x);
593  if (!world->use2D()) {
594  borderStep.z = (factor * move.getDirection().z); // 3D case
595  }
596 
597  if (policy == WRAP) {
598  borderStart.y = (0);
599  borderStart.x = (move.getStartPos().x + borderStep.x);
600  if (!world->use2D()) {
601  borderStart.z = (move.getStartPos().z + borderStep.z); // 3D case
602  }
603  }
604  break;
605 
606  case Z_SMALLER: // here we are definitely in 3D
607  factor = borderStep.z / move.getDirection().z;
608  borderStep.x = (factor * move.getDirection().x);
609  borderStep.y = (factor * move.getDirection().y);
610 
611  if (policy == WRAP) {
612  borderStart.z = (playgroundSizeZ());
613  borderStart.x = (move.getStartPos().x + borderStep.x);
614  borderStart.y = (move.getStartPos().y + borderStep.y);
615  }
616  break;
617 
618  case Z_BIGGER: // here we are definitely in 3D
619  factor = borderStep.z / move.getDirection().z;
620  borderStep.x = (factor * move.getDirection().x);
621  borderStep.y = (factor * move.getDirection().y);
622 
623  if (policy == WRAP) {
624  borderStart.z = (0);
625  borderStart.x = (move.getStartPos().x + borderStep.x);
626  borderStart.y = (move.getStartPos().y + borderStep.y);
627  }
628  break;
629 
630  default:
631  throw cRuntimeError("invalid state in goToBorder switch!");
632  break;
633  }
634 
635  EV_TRACE << "goToBorder: startPos: " << move.getStartPos().info() << " borderStep: " << borderStep.info() << " borderStart: " << borderStart.info() << " factor: " << factor << endl;
636 }

638 bool BaseMobility::handleIfOutside(BorderPolicy policy, Coord& stepTarget, Coord& targetPos, Coord& step, double& angle)
639 {
640  // where did the host leave the playground?
641  BorderHandling wo;
642 
643  // step to reach the border
644  Coord borderStep;
645 
646  wo = checkIfOutside(stepTarget, borderStep);
647 
648  // just return if the next step is not outside the playground
649  if (wo == NOWHERE) return false;
650 
651  EV_TRACE << "handleIfOutside:stepTarget = " << stepTarget.info() << endl;
652 
653  // new start position after the host reaches the border
654  Coord borderStart;
655  // new direction the host has to move to
656  Coord borderDirection;
657  // time to reach the border
658  simtime_t borderInterval;
659 
660  EV_TRACE << "old values: stepTarget: " << stepTarget.info() << " step: " << step.info() << " targetPos: " << targetPos.info() << " angle: " << angle << endl;
661 
662  // which border policy is to be followed
663  switch (policy) {
664  case REFLECT:
665  reflectIfOutside(wo, stepTarget, targetPos, step, angle);
666  break;
667  case WRAP:
668  wrapIfOutside(wo, stepTarget, targetPos);
669  break;
670  case PLACERANDOMLY:
671  placeRandomlyIfOutside(targetPos);
672  break;
673  case RAISEERROR:
674  break;
675  }
676 
677  EV_TRACE << "new values: stepTarget: " << stepTarget.info() << " step: " << step.info() << " angle: " << angle << endl;
678 
679  // calculate the step to reach the border
680  goToBorder(policy, wo, borderStep, borderStart);
681 
682  // calculate the time to reach the border
683  borderInterval = (borderStep.length()) / move.getSpeed();
684 
685  // calculate new start position
686  // NOTE: for WRAP this is done in goToBorder
687  switch (policy) {
688  case REFLECT: {
689  borderStart = move.getStartPos() + borderStep;
690  double d = stepTarget.distance(borderStart);
691  borderDirection = (stepTarget - borderStart) / d;
692  break;
693  }
694  case PLACERANDOMLY:
695  borderStart = targetPos;
696  stepTarget = targetPos;
697  break;
698 
699  case WRAP:
700  case RAISEERROR:
701  break;
702 
703  default:
704  throw cRuntimeError("unknown BorderPolicy");
705  break;
706  }
707 
708  EV_DEBUG << "border handled, borderStep: " << borderStep.info() << "borderStart: " << borderStart.info() << " stepTarget " << stepTarget.info() << endl;
709 
710  // create a border self message and schedule it appropriately
711  BorderMsg* bMsg = new BorderMsg("borderMove", MOVE_TO_BORDER);
712  bMsg->setPolicy(policy);
713  bMsg->setStartPos(borderStart);
714  bMsg->setDirection(borderDirection);
715 
716  scheduleAt(simTime() + borderInterval, bMsg);
717 
718  return true;
719 }
