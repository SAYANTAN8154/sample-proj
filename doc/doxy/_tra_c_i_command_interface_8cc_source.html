60 std::pair<uint32_t, std::string> TraCICommandInterface::getVersion()
61 {
62  TraCIConnection::Result result;
63  TraCIBuffer buf = connection.query(CMD_GETVERSION, TraCIBuffer(), &result);
64 
65  if (!result.success) {
66  ASSERT(buf.eof());
67  return std::pair<uint32_t, std::string>(0, "(unknown)");
68  }
69 
70  uint8_t cmdLength;
71  buf >> cmdLength;
72  uint8_t commandResp;
73  buf >> commandResp;
74  ASSERT(commandResp == CMD_GETVERSION);
75  uint32_t apiVersion;
76  buf >> apiVersion;
77  std::string serverVersion;
78  buf >> serverVersion;
79  ASSERT(buf.eof());
80 
81  return std::pair<uint32_t, std::string>(apiVersion, serverVersion);
82 }

95 std::pair<TraCICoord, TraCICoord> TraCICommandInterface::initNetworkBoundaries(int margin)
96 {
97  // query road network boundaries
98  TraCIBuffer buf = connection.query(CMD_GET_SIM_VARIABLE, TraCIBuffer() << static_cast<uint8_t>(VAR_NET_BOUNDING_BOX) << std::string("sim0"));
99  uint8_t cmdLength_resp;
100  buf >> cmdLength_resp;
101  uint8_t commandId_resp;
102  buf >> commandId_resp;
103  ASSERT(commandId_resp == RESPONSE_GET_SIM_VARIABLE);
104  uint8_t variableId_resp;
105  buf >> variableId_resp;
106  ASSERT(variableId_resp == VAR_NET_BOUNDING_BOX);
107  std::string simId;
108  buf >> simId;
109  uint8_t typeId_resp;
110  buf >> typeId_resp;
111  ASSERT(typeId_resp == getNetBoundaryType());
112  if (getNetBoundaryType() == TYPE_POLYGON) {
113  // Polygons can have an arbitrary number of tuples, so check that it is actually 2
114  uint8_t npoints;
115  buf >> npoints;
116  ASSERT(npoints == 2);
117  }
118  double x1;
119  buf >> x1;
120  double y1;
121  buf >> y1;
122  double x2;
123  buf >> x2;
124  double y2;
125  buf >> y2;
126  ASSERT(buf.eof());
127  EV_DEBUG << "TraCI reports network boundaries (" << x1 << ", " << y1 << ")-(" << x2 << ", " << y2 << ")" << endl;
128  TraCICoord nb1(x1, y1);
129  TraCICoord nb2(x2, y2);
130  connection.setNetbounds(nb1, nb2, margin);
131  return {nb1, nb2};
132 }

167  uint8_t cmdLength;

183  std::string objectId = nodeId;
184  ASSERT(objectId_r == objectId);
185  uint8_t resType_r;
186  buf >> resType_r;
187  uint8_t resultTypeId = TYPE_COLOR;
188  ASSERT(resType_r == resultTypeId);
189  buf >> res.red;
190  buf >> res.green;

338 {
339  if (travelTime >= 0) {

342  int32_t count = 2;
343  uint8_t edgeIdT = TYPE_STRING;

350  else {
351  uint8_t variableId = VAR_EDGE_TRAVELTIME;
352  uint8_t variableType = TYPE_COMPOUND;

356  TraCIBuffer buf = connection->query(CMD_SET_VEHICLE_VARIABLE, TraCIBuffer() << variableId << nodeId << variableType << count << edgeIdT << edgeId);
357  ASSERT(buf.eof());
358  }
359  {
360  uint8_t variableId = CMD_REROUTE_TRAVELTIME;
361  uint8_t variableType = TYPE_COMPOUND;
362  int32_t count = 0;
363  TraCIBuffer buf = connection->query(CMD_SET_VEHICLE_VARIABLE, TraCIBuffer() << variableId << nodeId << variableType << count);
364  ASSERT(buf.eof());

495 double TraCICommandInterface::getDistance(const Coord& p1, const Coord& p2, bool returnDrivingDistance)
496 {
497  uint8_t variable = DISTANCE_REQUEST;
498  std::string simId = "sim0";
499  uint8_t variableType = TYPE_COMPOUND;
500  int32_t count = 3;
501  uint8_t dType = static_cast<uint8_t>(returnDrivingDistance ? REQUEST_DRIVINGDIST : REQUEST_AIRDIST);
502 
503  // query road network boundaries
504 #if (VEINS_VERSION_MAJOR == 5)
505  TraCIBuffer buf = connection.query(CMD_GET_SIM_VARIABLE, TraCIBuffer() << variable << simId << variableType << count << connection.omnet2traci(p1) << connection.omnet2traci(p2) << dType);
506 #else
507  TraCIBuffer buf = connection.query(CMD_GET_SIM_VARIABLE, TraCIBuffer() << variable << simId << variableType << count << static_cast<uint8_t>(POSITION_2D) << connection.omnet2traci(p1) << static_cast<uint8_t>(POSITION_2D) << connection.omnet2traci(p2) << dType);
508 #endif
509  uint8_t cmdLength_resp;
510  buf >> cmdLength_resp;
511  uint8_t commandId_resp;
512  buf >> commandId_resp;
513  ASSERT(commandId_resp == RESPONSE_GET_SIM_VARIABLE);
514  uint8_t variableId_resp;
515  buf >> variableId_resp;
516  ASSERT(variableId_resp == variable);
517  std::string simId_resp;
518  buf >> simId_resp;
519  ASSERT(simId_resp == simId);
520  uint8_t typeId_resp;
521  buf >> typeId_resp;
522  ASSERT(typeId_resp == TYPE_DOUBLE);
523  double distance;
524  buf >> distance;
525 
526  ASSERT(buf.eof());
527 
528  return distance;
