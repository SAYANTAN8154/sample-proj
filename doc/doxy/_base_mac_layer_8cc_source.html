44 /**
45  * First we have to initialize the module from which we derived ours,
46  * in this case BaseLayer.
47  *
48  **/
49 void BaseMacLayer::initialize(int stage)
50 {
51  BaseLayer::initialize(stage);
52 
53  if (stage == 0) {
54  // get handle to phy layer
55  if ((phy = FindModule<MacToPhyInterface*>::findSubModule(getParentModule())) == nullptr) {
56  throw cRuntimeError("Could not find a PHY module.");
57  }
58 
59  headerLength = par("headerLength");
60  }
61  if (myMacAddr == LAddress::L2NULL()) {
62  // see if there is an addressing module available
63  // otherwise use NIC modules id as MAC address
64  AddressingInterface* addrScheme = FindModule<AddressingInterface*>::findSubModule(findHost());
65  if (addrScheme) {
66  myMacAddr = addrScheme->myMacAddr(this);
67  }
68  else {
69  const std::string addressString = par("address").stringValue();
70  if (addressString.empty() || addressString == "auto")
71  myMacAddr = LAddress::L2Type(getParentModule()->getId());
72  else
73  myMacAddr = 0;
74  }
75  registerInterface();
76  }
77 }

96 /**
97  * Encapsulates the received NetwPkt into a MacPkt and set all needed
98  * header fields.
99  **/
100 MacPkt* BaseMacLayer::encapsMsg(cPacket* netwPkt)
101 {
102  MacPkt* pkt = new MacPkt(netwPkt->getName(), netwPkt->getKind());
103  pkt->setBitLength(headerLength);
104 
105  // copy dest address from the Control Info attached to the network
106  // message by the network layer
107  cObject* cInfo = netwPkt->removeControlInfo();
108 
109  EV_TRACE << "CInfo removed, mac addr=" << getUpperDestinationFromControlInfo(cInfo) << endl;
110  pkt->setDestAddr(getUpperDestinationFromControlInfo(cInfo));
111 
112  // delete the control info
113  delete cInfo;
114 
115  // set the src address to own mac address (nic module getId())
116  pkt->setSrcAddr(myMacAddr);
117 
118  // encapsulate the network packet
119  pkt->encapsulate(netwPkt);
120  EV_TRACE << "pkt encapsulated\n";
121 
122  return pkt;
123 }

125 /**
126  * Redefine this function if you want to process messages from upper
127  * layers before they are send to lower layers.
128  *
129  * To forward the message to lower layers after processing it please
130  * use @ref sendDown. It will take care of anything needed
131  **/
132 void BaseMacLayer::handleUpperMsg(cMessage* mac)
133 {
134  ASSERT(dynamic_cast<cPacket*>(mac));
135  sendDown(encapsMsg(static_cast<cPacket*>(mac)));

138 /**
139  * This basic implementation just forwards all message that are
140  * broadcast (destAddr = L2BROADCAST) or destined for this node
141  * (destAddr = nic module getId()) to the network layer
142  *
143  * @sa sendUp
144  **/
145 
146 void BaseMacLayer::handleLowerMsg(cMessage* msg)
147 {
148  MacPkt* mac = static_cast<MacPkt*>(msg);
149  LAddress::L2Type dest = mac->getDestAddr();
150  LAddress::L2Type src = mac->getSrcAddr();
151 
152  // only foward to upper layer if message is for me or broadcast
153  if ((dest == myMacAddr) || LAddress::isL2Broadcast(dest)) {
154  EV_TRACE << "message with mac addr " << src << " for me (dest=" << dest << ") -> forward packet to upper layer\n";
155  sendUp(decapsMsg(mac));
156  }
157  else {
158  EV_TRACE << "message with mac addr " << src << " not for me (dest=" << dest << ") -> delete (my MAC=" << myMacAddr << ")\n";
159  delete mac;
160  }
161 }

163 void BaseMacLayer::handleLowerControl(cMessage* msg)
164 {
165  switch (msg->getKind()) {
166  case MacToPhyInterface::TX_OVER:
167  msg->setKind(TX_OVER);
168  sendControlUp(msg);
169  break;
170  default:
171  EV << "BaseMacLayer does not handle control messages of this type (name was " << msg->getName() << ")\n";
172  delete msg;
173  break;
174  }
