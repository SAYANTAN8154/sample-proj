29  * Simple Nakagami-m fading (based on a constant factor across all time and frequencies).
30  */
31 void NakagamiFading::filterSignal(Signal* signal)
32 {
33  auto senderPos = signal->getSenderPoa().pos.getPositionAt();
34  auto receiverPos = signal->getReceiverPoa().pos.getPositionAt();
35 
36  const double M_CLOSE = 1.5;
37  const double M_FAR = 0.75;
38  const double DIS_THRESHOLD = 80;
39 
40  EV_TRACE << "Add NakagamiFading ..." << endl;
41 
42  // get average TX power
43  // FIXME: really use average power (instead of max)
44  EV_TRACE << "Finding max TX power ..." << endl;
45  double sendPower_mW = signal->getMax();
46  EV_TRACE << "TX power is " << FWMath::mW2dBm(sendPower_mW) << " dBm" << endl;
47 
48  // get m value
49  double m = this->m;
50  {
51  const Coord senderPos2D(senderPos.x, senderPos.y);
52  const Coord receiverPos2D(receiverPos.x, receiverPos.y);
53  double d = senderPos2D.distance(receiverPos2D);
54  if (!constM) {
55  m = (d < DIS_THRESHOLD) ? M_CLOSE : M_FAR;
56  }
57  }
58 
59  // calculate average RX power
60  double recvPower_mW = (RNGCONTEXT gamma_d(m, sendPower_mW / 1000 / m)) * 1000.0;
61  if (recvPower_mW > sendPower_mW) {
62  recvPower_mW = sendPower_mW;
63  }
64  EV_TRACE << "RX power is " << FWMath::mW2dBm(recvPower_mW) << " dBm" << endl;
65 
66  // infer average attenuation
67  double factor = recvPower_mW / sendPower_mW;
68  EV_TRACE << "factor is: " << factor << " (i.e. " << FWMath::mW2dBm(factor) << " dB)" << endl;
69 
70  *signal *= factor;
71 }
