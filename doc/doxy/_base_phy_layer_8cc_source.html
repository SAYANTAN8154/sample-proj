65 void BasePhyLayer::initialize(int stage)
66 {
67 
68  ChannelAccess::initialize(stage);
69 
70  if (stage == 0) {
71  // if using sendDirect, make sure that messages arrive without delay
72 #if OMNETPP_BUILDNUM < 1506
73  gate("radioIn")->setDeliverOnReceptionStart(true);
74 #else
75  gate("radioIn")->setDeliverImmediately(true);
76 #endif
77 
78  upperLayerIn = findGate("upperLayerIn");
79  upperLayerOut = findGate("upperLayerOut");
80  upperControlOut = findGate("upperControlOut");
81  upperControlIn = findGate("upperControlIn");
82 
83  if (par("useNoiseFloor").boolValue()) {
84  noiseFloorValue = FWMath::dBm2mW(par("noiseFloor").doubleValue());
85  }
86  else {
87  noiseFloorValue = 0;
88  }
89  minPowerLevel = par("minPowerLevel").doubleValue();
90  minPowerLevel = FWMath::dBm2mW(minPowerLevel);
91 
92  recordStats = par("recordStats").boolValue();
93 
94  radio = initializeRadio();
95 
96  world = FindModule<BaseWorldUtility*>::findGlobalModule();
97  if (world == nullptr) {
98  throw cRuntimeError("Could not find BaseWorldUtility module");
99  }
100 
101  if (cc->hasPar("sat") && (minPowerLevel - FWMath::dBm2mW(cc->par("sat").doubleValue())) < -0.000001) {
102  throw cRuntimeError("minPowerLevel can't be smaller than the signal attenuation threshold (sat) in ConnectionManager. Please adjust your omnetpp.ini file accordingly.");
103  }
104 
105  initializeAnalogueModels(par("analogueModels").xmlValue());
106  initializeDecider(par("decider").xmlValue());
107  initializeAntenna(par("antenna").xmlValue());
108 
109  radioSwitchingOverTimer = new cMessage("radio switching over", RADIO_SWITCHING_OVER);
110  txOverTimer = new cMessage("transmission over", TX_OVER);
111  }
112 }

143 void BasePhyLayer::getParametersFromXML(cXMLElement* xmlData, ParameterMap& outputMap)
144 {
145  cXMLElementList parameters = xmlData->getElementsByTagName("Parameter");
146 
147  for (cXMLElementList::const_iterator it = parameters.begin(); it != parameters.end(); it++) {
148 
149  const char* name = (*it)->getAttribute("name");
150  const char* type = (*it)->getAttribute("type");
151  const char* value = (*it)->getAttribute("value");
152  if (name == nullptr || type == nullptr || value == nullptr) throw cRuntimeError("Invalid parameter, could not find name, type or value");
153 
154  std::string sType = type; // needed for easier comparision
155  std::string sValue = value; // needed for easier comparision
156 
157  cMsgPar param(name);
158 
159  // parse type of parameter and set value
160  if (sType == "bool") {
161  param.setBoolValue(sValue == "true" || sValue == "1");
162  }
163  else if (sType == "double") {
164  param.setDoubleValue(strtod(value, nullptr));
165  }
166  else if (sType == "string") {
167  param.setStringValue(value);
168  }
169  else if (sType == "long") {
170  param.setLongValue(strtol(value, nullptr, 0));
171  }
172  else {
173  throw cRuntimeError("Unknown parameter type: '%s'", sType.c_str());
174  }
175 
176  // add parameter to output map
177  outputMap[name] = param;
178  }

287 std::shared_ptr<Antenna> BasePhyLayer::initializeSampledAntenna1D(ParameterMap& params)
288 {
289  // get samples of the modeled antenna and put them in a vector
290  ParameterMap::iterator it = params.find("samples");
291  if (it == params.end()) {
292  throw cRuntimeError("BasePhyLayer::initializeSampledAntenna1D(): No samples specified for this antenna. \
293  Please adjust your xml file accordingly.");
294  }
295 
296  std::vector<double> values;
297  std::stringstream samplesStream(it->second.stringValue());
298  std::copy(std::istream_iterator<double>(samplesStream), std::istream_iterator<double>(), std::back_inserter(values));
299 
300  // get optional random offsets for the antenna's samples
301  std::string offsetType = "";
302  std::vector<double> offsetParams;
303  it = params.find("random-offsets");
304  if (it != params.end()) {
305  std::string buf;
306  std::stringstream offsetStream(it->second.stringValue());
307  offsetStream >> offsetType;
308  std::copy(std::istream_iterator<double>(offsetStream), std::istream_iterator<double>(), std::back_inserter(offsetParams));
309  }
310 
311  // get optional random rotation of the whole pattern
312  std::string rotationType = "";
313  std::vector<double> rotationParams;
314  it = params.find("random-rotation");
315  if (it != params.end()) {
316  std::string buf;
317  std::stringstream rotationStream(it->second.stringValue());
318  rotationStream >> rotationType;
319  std::copy(std::istream_iterator<double>(rotationStream), std::istream_iterator<double>(), std::back_inserter(rotationParams));
320  }
321 
322  return std::make_shared<SampledAntenna1D>(values, offsetType, offsetParams, rotationType, rotationParams, this->getRNG(0));
323 }

327 void BasePhyLayer::initializeAnalogueModels(cXMLElement* xmlConfig)
328 {
329 
330  if (xmlConfig == nullptr) {
331  throw cRuntimeError("No analogue models configuration file specified.");
332  }
333 
334  cXMLElementList analogueModelList = xmlConfig->getElementsByTagName("AnalogueModel");
335 
336  if (analogueModelList.empty()) {
337  throw cRuntimeError("No analogue models configuration found in configuration file.");
338  }
339 
340  // iterate over all AnalogueModel-entries, get a new AnalogueModel instance and add
341  // it to analogueModels
342  for (auto&& analogueModelData : analogueModelList) {
343  const char* name = analogueModelData->getAttribute("type");
344  const char* thresholdingFlag = analogueModelData->getAttribute("thresholding");
345 
346  if (name == nullptr) {
347  throw cRuntimeError("Could not read name of analogue model.");
348  }
349 
350  ParameterMap params;
351  getParametersFromXML(analogueModelData, params);
352 
353  auto newAnalogueModel = getAnalogueModelFromName(name, params);
354 
355  if (!newAnalogueModel) {
356  throw cRuntimeError("Could not find an analogue model with the name \"%s\".", name);
357  }
358 
359  // attach the new AnalogueModel to the AnalogueModelList
360  if (thresholdingFlag && std::string(thresholdingFlag) == "true") {
361  if (!newAnalogueModel->neverIncreasesPower()) {
362  throw cRuntimeError("Tried to instantiate analogue model \"%s\" with tresholding=true, but model does not support this.", name);
363  }
364  analogueModelsThresholding.push_back(std::move(newAnalogueModel));
365  }
366  else {
367  analogueModels.push_back(std::move(newAnalogueModel));
368  }
369 
370  EV_TRACE << "AnalogueModel \"" << name << "\" loaded." << endl;
371  }
372 }

376 void BasePhyLayer::handleMessage(cMessage* msg)
377 {
378 
379  // self messages
380  if (msg->isSelfMessage()) {
381  handleSelfMessage(msg);
382 
383  // MacPkts <- MacToPhyControlInfo
384  }
385  else if (msg->getArrivalGateId() == upperLayerIn) {
386  handleUpperMessage(msg);
387 
388  // controlmessages
389  }
390  else if (msg->getArrivalGateId() == upperControlIn) {
391  handleUpperControlMessage(msg);
392 
393  // AirFrames
394  }
395  else if (msg->getKind() == AIR_FRAME) {
396  handleAirFrame(static_cast<AirFrame*>(msg));
397 
398  // unknown message
399  }
400  else {
401  EV << "Unknown message received." << endl;
402  delete msg;
403  }
404 }

406 void BasePhyLayer::handleAirFrame(AirFrame* frame)
407 {
408  // TODO: ask jerome to set air frame priority in his UWBIRPhy
409  // ASSERT(frame->getSchedulingPriority() == airFramePriority());
410 
411  switch (static_cast<AirFrameState>(frame->getState())) {
412  case AirFrameState::start_receive:
413  handleAirFrameStartReceive(frame);
414  break;
415 
416  case AirFrameState::receiving: {
417  handleAirFrameReceiving(frame);
418  break;
419  }
420  case AirFrameState::end_receive:
421  handleAirFrameEndReceive(frame);
422  break;
423 
424  default:
425  throw cRuntimeError("Unknown AirFrame state: %d", frame->getState());
426  }

429 void BasePhyLayer::handleAirFrameStartReceive(AirFrame* frame)
430 {
431  EV_TRACE << "Received new AirFrame " << frame << " from channel." << endl;
432 
433  channelInfo.addAirFrame(frame, simTime());
434  ASSERT(!channelInfo.isChannelEmpty());
435 
436  if (usePropagationDelay) {
437  Signal& s = frame->getSignal();
438  simtime_t delay = simTime() - s.getSendingStart();
439  s.setPropagationDelay(delay);
440  }
441  ASSERT(frame->getSignal().getReceptionStart() == simTime());
442 
443  filterSignal(frame);
444 
445  if (decider && isKnownProtocolId(frame->getProtocolId())) {
446  frame->setState(static_cast<int>(AirFrameState::receiving));
447 
448  // pass the AirFrame the first time to the Decider
449  handleAirFrameReceiving(frame);
450 
451  // if no decider is defined we will schedule the message directly to its end
452  }
453  else {
454  Signal& signal = frame->getSignal();
455 
456  simtime_t signalEndTime = signal.getReceptionStart() + frame->getDuration();
457  frame->setState(static_cast<int>(AirFrameState::end_receive));
458 
459  sendSelfMessage(frame, signalEndTime);
460  }
461 }
462 
463 void BasePhyLayer::handleAirFrameReceiving(AirFrame* frame)
464 {
465 
466  Signal& signal = frame->getSignal();
467  simtime_t nextHandleTime = decider->processSignal(frame);
468 
469  ASSERT(signal.getDuration() == frame->getDuration());
470  simtime_t signalEndTime = signal.getReceptionStart() + frame->getDuration();
471 
472  // check if this is the end of the receiving process
473  if (simTime() >= signalEndTime) {
474  frame->setState(static_cast<int>(AirFrameState::end_receive));
475  handleAirFrameEndReceive(frame);
476  return;
477  }
478 
479  // smaller zero means don't give it to me again
480  if (nextHandleTime < 0) {
481  nextHandleTime = signalEndTime;
482  frame->setState(static_cast<int>(AirFrameState::end_receive));
483 
484  // invalid point in time
485  }
486  else if (nextHandleTime < simTime() || nextHandleTime > signalEndTime) {
487  throw cRuntimeError("Invalid next handle time returned by Decider. Expected a value between current simulation time (%.2f) and end of signal (%.2f) but got %.2f", SIMTIME_DBL(simTime()), SIMTIME_DBL(signalEndTime), SIMTIME_DBL(nextHandleTime));
488  }
489 
490  EV_TRACE << "Handed AirFrame with ID " << frame->getId() << " to Decider. Next handling in " << nextHandleTime - simTime() << "s." << endl;
491 
492  sendSelfMessage(frame, nextHandleTime);
493 }

495 void BasePhyLayer::handleAirFrameEndReceive(AirFrame* frame)
496 {
497  EV_TRACE << "End of Airframe with ID " << frame->getId() << "." << endl;
498 
499  simtime_t earliestInfoPoint = channelInfo.removeAirFrame(frame);
500

502  * of information in the ChannelInfo,
503  * since this time-point might have changed due to removal of
504  * the AirFrame
505  */
506  if (channelInfo.isChannelEmpty()) {
507  earliestInfoPoint = simTime();
508  }

511 void BasePhyLayer::handleUpperMessage(cMessage* msg)
512 {
513 
514  // check if Radio is in TX state
515  if (radio->getCurrentState() != Radio::TX) {
516  delete msg;
517  msg = nullptr;
518  throw cRuntimeError("Error: message for sending received, but radio not in state TX");
519  }
520 
521  // check if not already sending
522  if (txOverTimer->isScheduled()) {
523  delete msg;
524  msg = nullptr;
525  throw cRuntimeError("Error: message for sending received, but radio already sending");
526  }
527 
528  // build the AirFrame to send
529  ASSERT(dynamic_cast<cPacket*>(msg) != nullptr);
530 
531  unique_ptr<AirFrame> frame = encapsMsg(static_cast<cPacket*>(msg));
532 
533  // Prepare a POA object and attach it to the created Airframe
534  AntennaPosition pos = antennaPosition;
535  Coord orient = antennaHeading.toCoord();
536  frame->setPoa({pos, orient, antenna});
537 
538  // make sure there is no self message of kind TX_OVER scheduled
539  // and schedule the actual one
540  ASSERT(!txOverTimer->isScheduled());
541  sendSelfMessage(txOverTimer, simTime() + frame->getDuration());
542 
543  sendMessageDown(frame.release());
544 }

551 unique_ptr<AirFrame> BasePhyLayer::encapsMsg(cPacket* macPkt)
552 {
553  // the cMessage passed must be a MacPacket... but no cast needed here
554  // MacPkt* pkt = static_cast<MacPkt*>(msg);
555 
556  // ...and must always have a ControlInfo attached (contains Signal)
557  auto ctrlInfo = unique_ptr<cObject>(macPkt->removeControlInfo());
558  ASSERT(ctrlInfo);
559 
560  // create the new AirFrame
561  auto frame = createAirFrame(macPkt);
562 
563  // set the members
564  // set priority of AirFrames above the normal priority to ensure
565  // channel consistency (before any thing else happens at a time
566  // point t make sure that the channel has removed every AirFrame
567  // ended at t and added every AirFrame started at t)
568  frame->setSchedulingPriority(airFramePriority());
569  frame->setProtocolId(myProtocolId());
570  frame->setId(world->getUniqueAirFrameId());
571  frame->setChannel(radio->getCurrentChannel());
572  frame->setBitLength(0); // as the PHY might mix multiple different encoding rates (and things that might not be bits at all)
573 
574  frame->encapsulate(macPkt);
575 
576  // attach the spectrum-dependent Signal to the airFrame
577  attachSignal(frame.get(), ctrlInfo.get());
578 
579  // --- from here on, the AirFrame is the owner of the MacPacket ---
580  macPkt = nullptr;
581  EV_TRACE << "AirFrame encapsulated, length: " << frame->getBitLength() << "\n";
582 
583  return frame;
584 }

592 void BasePhyLayer::handleSelfMessage(cMessage* msg)
593 {
594 
595  switch (msg->getKind()) {

601

612 
613  default:

644 void BasePhyLayer::filterSignal(AirFrame* frame)
645 {
646  ASSERT(dynamic_cast<ChannelAccess* const>(frame->getArrivalModule()) == this);
647  ASSERT(dynamic_cast<ChannelAccess* const>(frame->getSenderModule()));
648  Signal& signal = frame->getSignal();
649 
650  // Extract position and orientation of sender and receiver (this module) first
651  const AntennaPosition receiverPosition = antennaPosition;
652  const Coord receiverOrientation = antennaHeading.toCoord();
653  // get POA from frame with the sender's position, orientation and antenna
654  POA& senderPOA = frame->getPoa();
655  const AntennaPosition senderPosition = senderPOA.pos;
656  const Coord senderOrientation = senderPOA.orientation;
657 
658  // add position information to signal
659  signal.setSenderPoa(senderPOA);
660  signal.setReceiverPoa({receiverPosition, receiverOrientation, antenna});
661 
662  // compute gains at sender and receiver antenna
663  double receiverGain = antenna->getGain(receiverPosition.getPositionAt(), receiverOrientation, senderPosition.getPositionAt());
664  double senderGain = senderPOA.antenna->getGain(senderPosition.getPositionAt(), senderOrientation, receiverPosition.getPositionAt());
665 
666  // add the resulting total gain to the attenuations list
667  EV_TRACE << "Sender's antenna gain: " << senderGain << endl;
668  EV_TRACE << "Own (receiver's) antenna gain: " << receiverGain << endl;
669  signal *= receiverGain * senderGain;
670 
671  // go on with AnalogueModels

722 {
723  Enter_Method_Silent();
724  ASSERT(radio);
725 
726  if (txOverTimer && txOverTimer->isScheduled()) {
727  EV_WARN << "Switched radio while sending an AirFrame. The effects this would have on the transmission are not simulated by the BasePhyLayer!";
728  }
729 
730  simtime_t switchTime = radio->switchTo(rs, simTime());
731 
732  // invalid switch time, we are probably already switching
733  if (switchTime < 0) return switchTime;
734 
735  // if switching is done in exactly zero-time no extra self-message is scheduled
736  if (switchTime == 0.0) {
737  // TODO: in case of zero-time-switch, send no control-message to mac!
738  // maybe call a method finishRadioSwitchingSilent()
739  finishRadioSwitching();
740  }
741  else {
742  sendSelfMessage(radioSwitchingOverTimer, simTime() + switchTime);
743  }
744 
745  return switchTime;
746 }
